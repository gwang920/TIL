#include <iostream>
#include <vector>
#include <queue>
using namespace std;

struct SHARK{
	bool isAlive=true;
	int y,x,dir;
	queue<pair<pair<int,int>,int>> sLoc;
	int dy[5][5],dx[5][5];
};

SHARK shark[42]; 
int board[21][21],res;
int dy[5]={0,-1,1,0,0},dx[5]={0,0,0,-1,1};
int N,M,K,time=0;

/*
	shark의 위치
	shark의 번호 
	shark의 냄새 크기 
	shark의 냄새 추적 벡터
	shark의 방향
	
	4방향 전부 막혀있는 경우 => 자신의 방향으로 이동한다? 
	
	1) 상어 이동 
		1-1 상어가 순차적으로 이동한다. 
		1-2 이동하면서 냄새를 두고간다. 
		1-3 격자가 겹치는 경우를 처리한다.
	2) 냄새가 없어지는 경우를 처리한다
		board=0 처리
	3) 1인 상어만 남았는지 체크 
*/

void moving(int num){
	bool flag=false;
	int y=shark[num].y,x=shark[num].x,dir=shark[num].dir;
	for(int i=1;i<=4;i++){
		int ny=y+shark[num].dy[dir][i];
		int nx=x+shark[num].dx[dir][i];
		if(ny<0 || nx<0 || ny>N-1 || nx>N-1) continue;
		if(board[ny][nx]<0) continue;
		if(board[ny][nx]==0 || board[ny][nx]>num){
			if(board[ny][nx]>num) shark[board[ny][nx]].isAlive=false;
			flag=true;
			shark[num].y=ny; shark[num].x=nx; shark[num].dir=i;
			shark[num].sLoc.push({{y,x},-(K+time)});
			board[y][x]=-(K+time);
			board[ny][nx]=num;
			break;
		}
	}
	if(!flag){
		int ny=y+dy[dir];
		int nx=x+dx[dir];
		shark[num].y=ny; shark[num].x=nx;
		shark[num].sLoc.push({{y,x},-(K+time)});
		board[y][x]=-(K+time);
		board[ny][nx]=num;
	}
}

void print(){
	cout << endl;
	for(int i=0;i<N;i++){
		for(int j=0;j<N;j++){
			cout << board[i][j];
		}
		cout << endl;
	}
}


bool check(){
	int cnt=0;
	for(int i=2;i<=M;i++){
		if(shark[i].isAlive==false) cnt++;
	}	
	if(shark[1].isAlive==true && cnt==M-1) return true;
	return false;
}

void game(){
	while(1){
		if(time>1000){
			res=-1; return;
		}
		if(check()){
			res=time; return;
		}
		for(int i=1;i<=M;i++){
		if(shark[i].isAlive==false) continue;
			moving(i);
		}
		for(int i=1;i<=M;i++){
			for(int j=0;j<shark[i].sLoc.size();j++){
				auto now=shark[i].sLoc.front();
				int y=now.first.first, x=now.first.second;
				int t=now.second;
				if(-t>time) break;
				if(-t==time){
					board[y][x]=0;
					shark[i].sLoc.pop();
				}
			}
		}
		time++;	
	}
}

void input(){
	cin >> N >> M >> K;
	for(int i=0;i<N;i++){
		for(int j=0;j<N;j++){
			cin >> board[i][j];
			shark[board[i][j]].y=i;shark[board[i][j]].x=j;		
		}
	}
	for(int i=1;i<=M;i++){
		int tmp;
		cin >> tmp;
		shark[i].dir=tmp;
	}
	
	for(int i=1;i<=M;i++){
		for(int j=1;j<=4;j++){
			int tmp;
			for(int k=1;k<=4;k++){
				cin >> tmp;
				shark[i].dy[j][k]=dy[tmp];
				shark[i].dx[j][k]=dx[tmp];				
			}
		}
	}
}

int main(){
	input();
	game();
	cout << res << endl;
	return 0;
}
