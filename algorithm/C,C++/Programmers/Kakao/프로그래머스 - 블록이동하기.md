# 프로그래머스 - 블록이동하기

- 카테고리 : 시뮬레이션/BFS

```
1) 실패코드
청소기 4개인덱스 동시 제어 실패


2) 청소기 첫 인덱스 두개만 고려하자.

성공 코드에서 처음에 방향회전시 <- ^ 방향은 고려하지 않았다.
							  |
회전시에도 4방향 전부 고려해야 한다.
```



# 실패

```c++
#include <string>
#include <vector>
#include <queue>
#include <iostream>
using namespace std;

struct INFO{
  int fy,fx,sy,sx,cnt,dir;  
};
vector<vector<int>> b;
queue<INFO> q;
bool visit[101][101][101][101]={0,};
const int dy[]={0,1,0,-1},dx[]={1,0,-1,0};
int answer = 0,N;
    
void bfs(){
    q.push({0,0,0,1,0,0});
    visit[0][0][0][1]=1;
    while(!q.empty()){
        auto now=q.front(); q.pop();
        int fy=now.fy,fx=now.fx,sy=now.sy,sx=now.sx,cnt=now.cnt,dir=now.dir;
        cout << fy << fx << ' ';
        if((fy==N-1 && fx==N-1) || (sy==N && sx==N-1)){
            answer=cnt; return;
        }
        for(int i=0;i<4;i++){
            int nfy=fy+dy[i],nfx=fx+dx[i];
            int nsy=sy+dy[i],nsx=sx+dx[i];
            if(nfy<0 || nfx<0 || nfy>N-1 || nfx>N-1) continue;
            if(nsy<0 || nsx<0 || nsy>N-1 || nsx>N-1) continue;
            if(visit[nfy][nfx][nsy][nsx]) continue;
            if(b[nfy][nfx] || b[nsy][nsx]) continue;
            visit[nfy][nfx][nsy][nsx]=1;
            q.push({nfy,nfx,nsy,nsx,cnt+1,dir});
        }
        if(dir==0){
            for(int i=0;i<2;i++){
              if(!b[fy+1][fx] && !b[sy+1][sx]){
                int nfy=sy+(1-i),nfx=sx,nsy=fy+i,nsx=fx;
                if(visit[nfy][nfx][nsy][nsx]) continue;
                visit[nfy][nfx][nsy][nsx]=1;
                q.push({nfy,nfx,nsy,nsx,cnt+1,1});
            }  
            }
            if(fy<1) continue;
            for(int i=0;i<2;i++){
              if(!b[fy-1][fx] && !b[sy-1][sx]){
                int nfy=sy,nfx=sx+(-1+i),nsy=fy,nsx=fx-i;
                if(visit[nfy][nfx][nsy][nsx]) continue;
                visit[nfy][nfx][nsy][nsx]=1;
                q.push({nfy,nfx,nsy,nsx,cnt+1,1});
            }  
            }
        }else{
            for(int i=0;i<2;i++){
              if(!b[fy][fx+1] && !b[sy][sx+1]){
                int nfy=sy,nfx=sx+(1-i),nsy=fy,nsx=fx+i;
                if(visit[nfy][nfx][nsy][nsx]) continue;
                visit[nfy][nfx][nsy][nsx]=1;
                q.push({nfy,nfx,nsy,nsx,cnt+1,0});
            }  
            }
            if(fx<1) continue;
            for(int i=0;i<2;i++){
              if(!b[fy][fx-1] && !b[sy][sx-1]){
                int nfy=sy,nfx=sx+(i-1),nsy=fy,nsx=fx-i;
                if(visit[nfy][nfx][nsy][nsx]) continue;
                visit[nfy][nfx][nsy][nsx]=1;
                q.push({nfy,nfx,nsy,nsx,cnt+1,0});
            }  
            }
        }
    }
}

int solution(vector<vector<int>> board) {
    b=board; N=board.size();
    bfs();
    return answer;
}
```

```c++
#include <string>
#include <vector>
#include <queue>
#include <iostream>
using namespace std;

struct INFO{
  int y,x,cnt,dir;  
};
vector<vector<int>> b;
queue<INFO> q;
bool visit[101][101][2];
const int dy[]={0,1,0,-1},dx[]={1,0,-1,0};
int answer = 0,N;
    
void bfs(){
    q.push({0,0,0,0});
    visit[0][0][0]=1;
    while(!q.empty()){
        auto now=q.front(); q.pop();
        int y=now.y,x=now.x,cnt=now.cnt,dir=now.dir;
        if((dir==0 && y==N-1 && x==N-2) || (dir==1 && y==N-2 && x==N-1)){
            answer=cnt; return;
        }
        for(int i=0;i<4;i++){
            int ny=y+dy[i],nx=x+dx[i];
            if(ny<0 || nx<0 || ny>N-1 || nx>N-1) continue;
            if(dir==0 && nx+1>N-1) continue;
            if(dir==1 && ny+1>N-1) continue;
            if(dir==0 && (b[ny][nx] || b[ny][nx+1])) continue;
            if(dir==1 && (b[ny][nx] || b[ny+1][nx])) continue;
            if(visit[ny][nx][dir]) continue;
            visit[ny][nx][dir]=1;
            q.push({ny,nx,cnt+1,dir});
        }
        if(dir==0){
            for(int t=0;t<2;t++){
                if(t==0){
                 if(y+1>N-1) continue;
                 if(b[y+1][x] || b[y+1][x+1]) continue;
                 for(int i=0;i<2;i++){
                    if(i==0){
                        if(visit[y][x][1]) continue;
                        visit[y][x][1]=1; q.push({y,x,cnt+1,1});
                    }else{
                        if(visit[y][x+1][1]) continue;
                        visit[y][x+1][1]=1; q.push({y,x+1,cnt+1,1});
                    }
                 }
                }else{
                    if(y-1<0) continue;
                    if(b[y-1][x] || b[y-1][x+1]) continue;
                    for(int i=0;i<2;i++){
                        if(i==0){
                            if(visit[y-1][x][1]) continue;
                            visit[y-1][x][1]=1; q.push({y-1,x,cnt+1,1});
                        }else{
                            if(visit[y-1][x+1][1]) continue;
                            visit[y-1][x+1][1]=1; q.push({y-1,x+1,cnt+1,1});
                        }
                    }
                    
                }
            }
            
        }else{
            for(int t=0;t<2;t++){
                if(t==0){
                    if(x+1>N-1) continue;
                    if(b[y][x+1] || b[y+1][x+1]) continue;
                   for(int i=0;i<2;i++){
                   if(i==0){
                     if(visit[y][x][0]) continue;
                        visit[y][x][0]=1; q.push({y,x,cnt+1,0});
                  }else{
                       if(visit[y+1][x][0]) continue;
                       visit[y+1][x][0]=1; q.push({y+1,x,cnt+1,0});
                    }
              }
                }else{
                    if(x-1<0) continue;
                    if(b[y][x-1] || b[y+1][x-1]) continue;
                    for(int i=0;i<2;i++){
                    if(i==0){
                     if(visit[y][x-1][0]) continue;
                        visit[y][x-1][0]=1; q.push({y,x-1,cnt+1,0});
                  }else{
                       if(visit[y+1][x-1][0]) continue;
                       visit[y+1][x-1][0]=1; q.push({y+1,x-1,cnt+1,0});
                    }
                    }
                }
            }

        }
    }
}

int solution(vector<vector<int>> board) {
    b=board; N=board.size();
    bfs();
    return answer;
}
```

```c++
#include <string>
#include <vector>
#include <queue>
#include <iostream>
using namespace std;
int answer = 987654321;
bool visit[101][101][2]; 

struct INFO{
    int y,x,dir,time;
};

const int dy[]={0,1,0,-1};
const int dx[]={1,0,-1,0};

void bfs(vector<vector<int>> b){
    queue<INFO> q;
    q.push({0,0,0,0});
    
    while(!q.empty()){
        auto now=q.front(); q.pop();
        int y=now.y, x=now.x,dir=now.dir,time=now.time;
        if(dir==0 && y==b.size()-1 && x==b.size()-2){
            answer=min(answer,time); continue; 
        }
        else if(dir==1 && y==b.size()-2 && x==b.size()-1){
            answer=min(answer,time); continue; 
        }
        for(int i=0;i<4;i++){
            int ny=y+dy[i],nx=x+dx[i];
            if(ny<0 || nx<0) continue;
            if(visit[ny][nx][dir]) continue;
            if(dir==0){
                  if(ny>b.size()-1 || nx+1>b.size()-1) continue;
                  if(b[ny][nx+1] || b[ny][nx]) continue;  
            } 
            else if(dir==1){
                  if(nx>b.size()-1 || ny+1>b.size()-1) continue;
                  if(b[ny+1][nx] || b[ny][nx]) continue;  
            } 
            visit[ny][nx][dir]=1;
            q.push({ny,nx,dir,time+1});
        }
        
        if(dir==0){
            if(y-1>=0 && !b[y-1][x] && !b[y-1][x+1]){
                if(!visit[y-1][x][1]) q.push({y-1,x,1,time+1});
                if(!visit[y-1][x+1][1]) q.push({y-1,x+1,1,time+1});
                visit[y-1][x][1]=1; visit[y-1][x+1][1]=1;
            }
            if(y+1<=b.size()-1 && !b[y+1][x] && !b[y+1][x+1]){
                if(!visit[y][x][1])  q.push({y,x,1,time+1});
                if(!visit[y][x+1][1]) q.push({y,x+1,1,time+1});
                visit[y][x][1]=1; visit[y][x+1][1]=1;
            }
        }
        if(dir==1){
            if(x-1>=0 && !b[y][x-1] && !b[y+1][x-1]){
                if(!visit[y][x-1][0]) q.push({y,x-1,0,time+1}); 
                if(!visit[y+1][x-1][0]) q.push({y+1,x-1,0,time+1});
                visit[y][x-1][0]=1; visit[y+1][x-1][0]=1;
            }
            if(x+1<=b.size()-1 && !b[y][x+1] && !b[y+1][x+1]){
                if(!visit[y][x][0]) q.push({y,x,0,time+1});
                if(!visit[y+1][x][0]) q.push({y+1,x,0,time+1});
                visit[y][x][0]=1; visit[y+1][x][0]=1;
            }
        }
    }
}

int solution(vector<vector<int>> board) {
    bfs(board);
    return answer;
}
```
