# 실패

```c++
#include <iostream>
#include <vector>
#define MAX 42
using namespace std;

vector<pair<int,int>> v;
int N,M,K,map[MAX][MAX]; 
int sticker[101][MAX][MAX];

void rotate(int num,int by,int bx){
	int temp[MAX][MAX]={0,};
	for(int i=1;i<=by;i++){
		for(int j=1;j<=bx;j++){
				temp[j][N+1-i]=sticker[num][i][j];
		}
	}
	for(int i=1;i<=12;i++){
		for(int j=1;j<=12;j++){
			sticker[num][i][j]=temp[i][j];			
		}
	}
	v[num-1].first=bx; v[num-1].second=by;
}

void attach(int t,int y,int x){
	for(int i=y;i<=y+v[t-1].first;i++){
		for(int j=x;j<=x+v[t-1].second;j++){
			map[i][j]=sticker[t][i+1-y][j+1-x];		
		}
	}
}


bool check(int t,int y,int x){
	for(int i=y;i<=y+v[t-1].first;i++){
		for(int j=x;j<=x+v[t-1].second;j++){
			if(sticker[t][i+1-y][j+1-x] && map[i][j]) return false;	
			if(i<1 || j<1 || i>N || j>M) return false;	
		}
	}
	return true;
}

void game(){
	for(int t=1;t<=v.size();t++){
		// 시작점 부터 가능한 경우 모두
		for(int rot=0;rot<4;rot++){
			int by=v[t-1].first,bx=v[t-1].second;
			bool flag=false; 
			for(int i=1;i+by<=N;i++){
				if(flag) break;
				for(int j=1;j+bx<=M;j++){
					if(flag) break;
					if(check(t,i,j)){
						flag=true;
						attach(t,i,j);
					}		
				}
			}
			if(flag) break;
			if(!flag) rotate(t,by,bx);	
		}
	}
}

int main(){
	cin >> N >> M >> K;
	for(int t=1;t<=K;t++){
		int a,b;
		cin >> a >> b;
		v.push_back({a,b});
		for(int i=1;i<=a;i++){
			for(int j=1;j<=b;j++){
				cin >> sticker[t][i][j];
			}
		}
	}
	rotate(2,2,5);
	rotate(2,2,5);
//	game();
	int res=0;
	cout << endl;
	for(int i=1;i<=N;i++){
		for(int j=1;j<=M;j++){
			cout << sticker[2][i][j];
			if(map[i][j]) res++;
		}
		cout << endl;
	}
	cout << res << endl;
	return 0;
}
```

# 성공

```c++
#include <iostream>
#define MAX 42
using namespace std;

int board[MAX][MAX],res;
int sticker[12][12];
int N,M,K,r,c;

void rotate(){
	int tmp[12][12];
	for(int i=1;i<=r;i++)
		for(int j=1;j<=c;j++)
			tmp[i][j]=sticker[i][j];
			
	for(int i=1;i<=c;i++)
		for(int j=1;j<=r;j++)
			sticker[i][j]=tmp[r+1-j][i];
			
	swap(r,c);				
}

bool rotate_check(){
	for(int i=1;i+r<=N+1;i++){
		for(int j=1;j+c<=M+1;j++){
			bool flag=true;
			for(int y=1;y<=r;y++){
				if(!flag) break;
				for(int x=1;x<=c;x++){
					if(board[i+y-1][j+x-1]==1 && sticker[y][x]==1){
						flag=false; break;	
					} 
				}
			}
			if(flag){
				for(int y=1;y<=r;y++){
					for(int x=1;x<=c;x++){
						if(sticker[y][x]) board[i+y-1][j+x-1]=1;
					}
				}		
				return true;
			}
		}
	}
	return false;
}


int main(){
	
	cin >> N >> M >> K;
	while(K--){
		cin >> r >> c;
		for(int i=1;i<=r;i++)
			for(int j=1;j<=c;j++)
				cin >> sticker[i][j];
		if(rotate_check()) continue;
		rotate();
		if(rotate_check()) continue;
		rotate();
		if(rotate_check()) continue;
		rotate();
		if(rotate_check()) continue;
		rotate();		
	}
	for(int i=1;i<=N;i++)
		for(int j=1;j<=M;j++)
			res+=board[i][j]; 
	cout << res << endl;		
	return 0;
}
```
