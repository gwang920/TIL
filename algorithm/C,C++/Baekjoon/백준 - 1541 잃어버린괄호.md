# 실패

```c++
#include <iostream>
#include <vector>
using namespace std;

// 10-20+30-10+50-20+30+40+50
// 10-50-60-140
string str;

vector<int> num;
vector<char> oper;

void print(){
	for(int i=0;i<num.size();i++) cout << num[i] << ' ';
	cout << endl;
	for(int i=0;i<oper.size();i++) cout << oper[i] << ' ';
}
// 55-50+40
void go(){
	
	vector<int> back_num;
	vector<char> back_oper;
	
	
	for(int i=0;i<oper.size();i++){
		if(oper[i]=='-'){
			back_oper.push_back(oper[i]);
			int sum=num[i+1];
			for(int j=i+1;j<oper.size();j++){
				if(back_oper[j]=='-'){
					i=j; break;	
				}
				sum+=num[j+1];
			}	
			back_num.push_back(sum); 
		}else{
			continue;
		}
	}
	
	num=back_num; oper=back_oper;
	print();

	
}

void init(){
	string tmp="";
	for(int i=0;i<str.length();i++){
		if(str[i]=='+' || str[i]=='-'){
			if(i==0){
				int sum=0;
				for(int j=i+1;j<str.length();j++){
					if(str[j]=='-' || str[j]=='+'){
						if(str[j]=='-'){
							num.push_back(sum);
							i=j;
							break;	
						}
					}else{
						sum+=(int)str[j];
					} 
				}
			}
			oper.push_back(str[i]);
			if(i!=0) num.push_back(stoi(tmp));
			tmp="";
		}else{
			tmp+=str[i];
		}
		
		if(i==str.length()-1) num.push_back(stoi(tmp));
	}
	print();
	cout << endl;
}

void input(){
	cin >> str;
}

int main(){
	input();
	init();
	go();
	return 0;
}
```
