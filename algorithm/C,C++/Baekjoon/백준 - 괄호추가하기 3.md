```c++
#include <iostream>
#include <vector>

using namespace std;
string str;
int N,answer;

vector<string> num;
vector<char> cmd;

string calc(string a,string b,char op){
	int n=0;
	if(op=='+') n=stoi(a)+stoi(b); 
	if(op=='*') n=stoi(a)*stoi(b);
	if(op=='-') n=stoi(a)-stoi(b);
	cout << n << endl;
	return to_string(n);
}

void div(vector<string> n,vector<char> c,int cnt){
	if(N==cnt-1){
		cout << "ok";
		int res=stoi(n[0]);
		answer=max(answer,res);
		return;
	}
	for(int i=0;i<c.size();i++){
		vector<string> newn;
		vector<char> newc;
		string nnum=calc(n[i],n[i+1],c[i]);
		for(int j=0;j<c.size();j++){
			if(j==i){
				newn.push_back(nnum);
				continue;	
			} 
			newc.push_back(c[i]);
			if(j==i+1) continue;
			newn.push_back(n[j]);
		}
		
		div(newn,newc,cnt+1);
	}
}

void split(string str){
	for(int i=0;i<str.length();i++){
		if(str[i]=='+' || str[i]=='*' || str[i]=='-'){
			cmd.push_back(str[i]);
		}else{
			string s="";
			s+=str[i];
			num.push_back(s);
		}
	}
}

int main(){
	cin >> N >> str;
	split(str);
	div(num,cmd,0);
	cout << answer << endl;
	return 0;
}
```
