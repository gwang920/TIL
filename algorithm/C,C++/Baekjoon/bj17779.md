#include <iostream>
#include <algorithm>
#define MAX 21

using namespace std;


int N;
int map[MAX][MAX];
int res=987654321;
int pop[5];
int sum;
void calc(){
	int minn=987654321; int maxi=0;
	for(int i=0;i<5;i++){
			minn=min(minn,pop[i]);
			maxi=max(maxi,pop[i]);
	}

	res=min(res,maxi-minn);
}

void move(int d1,int d2){
	
	if(d1+d2>=N) return;
	
	for(int i=d1+1;i+d2<=N;i++){
		for(int j=1;j+d1+d2<=N;j++){
			
			pop[0]=0; int cnt=0; int std=i-d1;
			for(int l=1;l<=i;l++){
				if(l>=std) cnt++;
				for(int k=1;k<=j+d1-cnt;k++){
					pop[0]+=map[l][k];
				}
			}
			
			pop[1]=0; cnt=1; std=i-d1+1;
			for(int l=1;l<=i-(d1-d2);l++){
				if(l>=std) cnt++;
				for(int k=j+d1+cnt;k<=N;k++){
					pop[1]+=map[l][k];
				}
			}
			
			pop[2]=0; cnt=0; std=i+d2;
			for(int l=i;l<=N;l++){
				for(int k=1;k<j+cnt;k++){
					if(l==i && j==1) continue;
					pop[2]+=map[l][k];
				}
				if(l<std) cnt++;
			}
			
			
			pop[3]=0; cnt=0; std=i+d2;
			for(int l=i+abs(d1-d2)+1;l<=N;l++){
				for(int k=j+d1+d2-cnt;k<=N;k++){
					pop[3]+=map[l][k];
				}
				if(l<=std) cnt++;
			}
			
			int sy=i;int sx=j; int boundary=map[sy][sx];
			for(int l=0;l<d1;l++){
				sy-=1;sx+=1;
				boundary+=map[sy][sx];
			}
			for(int k=0;k<d2;k++){
				sy+=1; sx+=1;
				boundary+=map[sy][sx];
			}
			sy=i; sx=j;
			for(int l=0;l<d2;l++){
				sy+=1;sx+=1;
				boundary+=map[sy][sx];
			}
			for(int k=0;k<d1-1;k++){
				sy-=1;sx+=1;
				boundary+=map[sy][sx];
			}
			
			pop[4]=sum-(pop[0]+pop[1]+pop[2]+pop[3]+boundary);
			calc();
		}
	}
	
	move(d1,d2+1);
	move(d1+1,d2);
		
}


int main(){
	cin >> N;
	
	for(int i=1;i<=N;i++){
		for(int j=1;j<=N;j++){
			cin >> map[i][j];
			sum+=map[i][j];
		}
	}
	
	move(1,1);
	cout << res << endl;
	return 0;
}
