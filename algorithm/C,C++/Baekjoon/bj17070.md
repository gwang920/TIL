- BFS로 짰더니 답은 맞지만 메모리, 시간이 이상적이지 않다.

# 성공 (BFS)

```C++
#include <iostream>
#include <queue>
#define MAX 16

using namespace std;

int N;
int map[MAX][MAX];
int answer=0;

const int dy[]={0,1,1};
const int dx[]={1,0,1};

struct pipe{
	int first_y,first_x,end_y,end_x,dir;
};
queue<pipe> q;

void move(){
	
	q.push({0,1,0,0,0});
	
	while(!q.empty()){
		auto now=q.front(); q.pop();
		int f_y=now.first_y; int f_x=now.first_x;
		int e_y=now.end_y; int e_x=now.end_x;
		int d=now.dir;
		
		if(f_y==N-1 && f_x==N-1) answer++;
		
		for(int i=0;i<3;i++){
			
			if(d==0){
				if(i==1) continue;
			}
			
			if(d==1){
				if(i==0) continue;
			}
			
			int n_f_y=f_y+dy[i]; int n_f_x=f_x+dx[i];
			int n_e_y=e_y+dy[d]; int n_e_x=e_x+dx[d];
			int n_d=i;
			if(n_f_y>N-1 || n_f_x>N-1 || map[n_f_y][n_f_x]==1) continue;
			if(i==2){
				if(map[n_f_y-1][n_f_x] || map[n_f_y][n_f_x-1]) continue;
			}
			q.push({n_f_y,n_f_x,n_e_y,n_e_x,n_d});
		}
	}
	
}

int main(){
	
	cin >> N;
	for(int i=0;i<N;i++)
		for(int j=0;j<N;j++)
			cin >> map[i][j];
			
	move();
	cout << answer << endl;
	
	return 0;
}
```



# 추가 (DP - 참고)

```c++
#include <iostream>

using namespace std;

int dp_dig[17][17];
int dp_right[17][17];
int dp_down[17][17];
int map[17][17];

int num;
string form;

int main() {
	cin >> num;
	for (int i = 1; i <= num; i++) {
		for (int j = 1; j <= num; j++) {
			cin >> map[i][j];
		}
	}

	dp_right[1][2] = 1;

	for (int i = 1; i <= num; i++) {
		for (int j = 2; j <= num; j++) {
			if (map[i][j] == 1) continue;
			if (i == 1 && j == 2) continue;

			dp_right[i][j] = dp_right[i][j - 1] + dp_dig[i][j - 1];
			if(map[i-1][j] != 1 && map[i][j-1] != 1)dp_dig[i][j] = dp_dig[i-1][j - 1] + dp_right[i-1][j - 1]  +dp_down[i-1][j-1];
			dp_down[i][j] = dp_down[i-1][j] + dp_dig[i-1][j];
		}
	}
	int res = dp_right[num][num] + dp_down[num][num] + dp_dig[num][num];
	cout << res;
}
```
