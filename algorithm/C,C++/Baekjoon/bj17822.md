# 백준 - 17822 원판돌리기

- 카테고리 : 시뮬레이션

```
1) ki의 배수의원판을 회전시킨다
2) 회전을 마친후 인접구역이 같으면 -1로 갱신한다.
3) T번의 반복 후 모든 원판원소의 합을 구한다.
```



- 배열 시계/반시계 방향 회전시 단순 for문을 사용하면 index 설계가 어렵다

```c++
CASE 1 : for문 설계

for(int k=1;k<=c;k++){
	tmp[j][k]=cir[j][M-c+k-1];
}
for(int k=M;k>c;k--){
	tmp[j][k]=cir[j][k-c];
}
for(int k=1;k<=M;k++){
	cir[j][k]=tmp[j][k];
}

CASE 2 : vector활용한 for문

vector<int> tmp;
if(d==0){
	for(int k=M-c+1;k<=M;k++){
		tmp.push_back(cir[j][k]);
	}
	for(int k=1;k<=M-c;k++){
		tmp.push_back(cir[j][k]);
	}
	for(int k=1;k<=M;k++){
		cir[j][k]=tmp[k-1];
   }
}
```

# 실패

```c++
#include <iostream>
#include <vector>

using namespace std;

vector<pair<pair<int,int>,int>> v;

int N,M,T;
int cir[51][51];
int ans=0;
int main(){
	
	cin >> N >> M >> T;
	
	for(int i=1;i<=N;i++){
		for(int j=1;j<=M;j++){
			cin >> cir[i][j];
		}
	}
	
	for(int i=0;i<T;i++){
		int x,d,k;
		cin >> x >> d >> k;
		v.push_back({{x,d},k});
	}
	
	for(int i=0;i<v.size();i++){
		int x=v[i].first.first; int d=v[i].first.second; int c=v[i].second;
		c=c%M;
		for(int j=1;j<=N;j++){
			if(j%x==0){
					vector<int> tmp;
					if(d==0){
						for(int k=M-c+1;k<=M;k++){
							tmp.push_back(cir[j][k]);
						}
						for(int k=1;k<=M-c;k++){
							tmp.push_back(cir[j][k]);
						}
						for(int k=1;k<=M;k++){
							cir[j][k]=tmp[k-1];
						}
					}
					if(d==1){
						for(int k=c+1;k<=M;k++){
							tmp.push_back(cir[j][k]);
						}
						for(int k=1;k<=c;k++){
							tmp.push_back(cir[j][k]);
						}
						for(int k=1;k<=M;k++){
							cir[j][k]=tmp[k-1];
						}
					}
			}
		}
		
		bool flag=false;
		int visited[51][51]={0,};
		
		for(int k=1;k<=M;k++){
			for(int j=1;j<N;j++){
				if(cir[j][k]==-1) continue;
				if(cir[j][k]==cir[j+1][k]){
					visited[j][k]=1;visited[j+1][k]=1;
				}
			}
		}
		
		for(int j=1;j<=N;j++){
			for(int k=1;k<=M;k++){
				if(cir[j][k]==-1) continue;
				if(k==M){
					if(cir[j][k]==cir[j][1]){
						visited[j][k]=1; visited[j][1]=1;
						break;
					}
				}
				if(cir[j][k]==cir[j][k+1]){  // if가아닌 else if를 써야한다.
					visited[j][k]=1; visited[j][k+1]=1;
				}
			}
		}

				for(int j=1;j<=N;j++){
					for(int k=1;k<=M;k++){
						if(visited[j][k]){
							cir[j][k]=-1;flag=true;
						}
					}
				}

	
		double avg=0.0;int cnt=0;
		if(flag==false){
			for(int j=1;j<=N;j++){
				for(int k=1;k<=M;k++){
					if(cir[j][k]==-1) continue;
					avg+=(double)cir[j][k];
					cnt++;
				}
			}
			
			avg/=(double)cnt;
			for(int j=1;j<=N;j++){
				for(int k=1;k<=M;k++){
					if(cir[j][k]==-1) continue;
					if((double)cir[j][k]<avg){
						cir[j][k]++;
					}
					else if((double)cir[j][k]>avg){
						cir[j][k]--;
					}
				}
			}
		}
	}
	for(int i=1;i<=N;i++){
		for(int j=1;j<=M;j++){
			if(cir[i][j]==-1) continue;
			ans+=cir[i][j];
		}
	}
	cout << ans << endl;
	return 0;
}
```
