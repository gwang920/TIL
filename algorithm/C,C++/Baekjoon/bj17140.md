```c++
#include <iostream>
#include <queue>

using namespace std;
priority_queue<pair<int,int>> pq;

int main(){
	
	pq.push({2,1});
	pq.push({2,2});
	pq.push({1,3});
	pq.push({1,4});
	pq.push({3,5});
	int size=pq.size();
	for(int i=0;i<size;i++){
		cout << pq.top().first << " " << pq.top().second << endl;
		pq.pop();	
	}
	
	
	return 0;
}
```

```c++
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

vector<vector<int>> v;
int r,c,k;
int ans=0;

void set(int idx,int size){
	if(idx==0){
		for(int i=0;i<v.size();i++){
			while(v[i].size()<size){
				v[i].push_back(0);
			}	
		}
	}
	if(idx==1){
		for(int i=0;i<v.size();i++){
				if(v[i].size()==v[0].size()) continue;
				while(v[i].size()<v[0].size()){
					v[i].push_back(0);
				}
		}
	}
}

void count(int idx,int dir,int size){
	
	if(dir==0){
		int max=*max_element(v[idx].begin(),v[idx].end());
		int cnt[max+1]={0,};
		int max_count=0;
		for(int i=0;i<v[idx].size();i++){
				cnt[v[idx][i]]++;
				if(cnt[v[idx][i]]>max_count){
					max_count=cnt[v[idx][i]];
				}
		}
		vector<int> arr;
		for(int i=1;i<=max_count;i++){
			for(int j=1;j<=max;j++){
				if(cnt[j]==i){
					arr.push_back(j); arr.push_back(i);
				}
			}
		}
		if(v[idx].size()>=arr.size()){
			for(int i=0;i<arr.size();i++){
				v[idx][i]=arr[i];
			}
			for(int i=arr.size();i<v[idx].size();i++){
				v[idx][i]=0;
			}
		}else{
			for(int i=0;i<v[idx].size();i++){
			v[idx][i]=arr[i];
			}
			for(int i=v[idx].size();i<arr.size();i++){
				v[idx].push_back(arr[i]);
			}
		}
		
		if(v[idx].size()>100){
			v[idx].erase(v[idx].begin()+100,v[idx].end());
		}
		
	}
	if(dir==1){
		int max=0;
		for(int i=0;i<size;i++){
			if(max<v[i][idx]){
				max=v[i][idx];
			}
		}
		int cnt[max+1]={0,};
		int max_count=0;
		for(int i=0;i<size;i++){
				cnt[v[i][idx]]++;
				if(cnt[v[i][idx]]>max_count){
					max_count=cnt[v[i][idx]];
				}
		}
		vector<int> arr;
		for(int i=1;i<=max_count;i++){
			for(int j=1;j<=max;j++){
				if(cnt[j]==i){
					arr.push_back(j); arr.push_back(i);
				}
			}
		}
		if(size>=arr.size()){
			for(int i=0;i<arr.size();i++){
				v[i][idx]=arr[i];
			}
			for(int i=arr.size();i<size;i++){
				v[i][idx]=0;
			}
		}else{
			if(!idx){
				for(int i=0;i<size;i++){
					v[i][idx]=arr[i];
				}
				
				for(int i=size;i<arr.size();i++){
					v.push_back({arr[i]});
				}
			}else{
				for(int i=0;i<size;i++){
					v[i][idx]=arr[i];
				}
				for(int i=size;i<arr.size();i++){
					v[i].push_back({arr[i]});
				}
			}
		}
		
		if(v.size()>100){
			v.erase(v.begin()+100,v.end());
		}
	}
}

void calc(){
	while(1){
		if(v[r-1][c-1]==k) return;
		if(v[r-1][c-1]>100){
			ans=-1;
			return;	
		} 
		ans++;
		if(ans==10) return;
		int max_size=0;
		int row=v.size(),cul=v[0].size();
		if(row>=cul){
			for(int i=0;i<v.size();i++){
				count(i,0,0);
				if(max_size<v[i].size()){
					max_size=v[i].size();
				}
			}
			set(0,max_size);
		}
		else if(row<cul){
			for(int i=0;i<v[0].size();i++){
				count(i,1,row);
			}
			set(1,v.size());
			}
		}
}

int main(){
	
	cin >> r >> c >> k;
	
	for(int i=0;i<3;i++){
		int a,b,c;
		cin >> a >> b >> c;
		v.push_back({a,b,c});
	}
	calc();
	cout << endl;
	for(int i=0;i<v.size();i++){
		for(int j=0;j<v[i].size();j++){
			cout << v[i][j];
		}
		cout << endl;
	}
	cout << ans << endl;
	
	return 0;
}
```

#  성공

```c++
#include <iostream>
#include <queue>
#include <algorithm>
#include <memory.h>
using namespace std;

int r,c,k,n=3,m=3;
int cnt[101],map[101][101];
int ans=-1;
void solve(){
	
	for(int t=0;t<101;t++){
		if(map[r][c]==k){
			ans=t;
			return;
		}
		if(n>=m){
			for(int i=1;i<=n;i++){
				priority_queue<pair<int, int>> pq;
                memset(cnt, 0, sizeof(cnt));
				for(int j=1;j<=m;j++){
					if(map[i][j]){
						cnt[map[i][j]]++;
						map[i][j]=0;	
					}
				}
				for(int j=1;j<=100;j++){
					if(cnt[j]) pq.push({-cnt[j],-j});
				}
				int len=pq.size()*2;
				m=max(m,len);
				
				for(int j=1;j<=len;j+=2){
					auto now=pq.top(); pq.pop();
					map[i][j]=-now.second;
					map[i][j+1]=-now.first;
				}
			}
		}
		else if(n<m){
			for(int i=1;i<=m;i++){
				priority_queue<pair<int, int>> pq;
                memset(cnt, 0, sizeof(cnt));
				for(int j=1;j<=n;j++){
					if(map[j][i]){
						cnt[map[j][i]]++;
						map[j][i]=0;
					}
				}
				for(int j=1;j<=100;j++){
					if(cnt[j]){
						pq.push({-cnt[j],-j});
					}
				}
				int len=pq.size()*2;
				n=max(n,len);
				for(int j=1;j<=len;j+=2){
					auto now=pq.top(); pq.pop();
					map[j][i]=-now.second;
					map[j+1][i]=-now.first;
				}	
			}
		}
	}
}

int main(){
	cin >> r >> c >> k;
	
	for(int i=1;i<=3;i++){
		for(int j=1;j<=3;j++){
			cin >> map[i][j];
		}
	}
	solve();
	cout << ans << endl;
	return 0;
}
```
