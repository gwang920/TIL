- 치킨배달 거리 계산 간단하게 하는 방법?

  ```
  첫 코드에서 bfs로 가장 짧은거리를 탐색하려했다.
  => 시간 복잡도가 너무 커진다
  
  두번째 코드에서 벡터에 치킨집, 가정집 위치를 저장해서
  이에 대한 최소값을 갱신하면서 하나씩 저장을 하는 방법으로도 충분했다.
  ```

  

# 성공

```c++
#include <iostream>
#include <queue>
#include <vector>
#include <algorithm>
#define MAX 51

using namespace std;

queue<pair<int,int>> q;
vector<pair<int,int>> v;

const int dx[]={0,1,0,-1};
const int dy[]={1,0,-1,0};

int N,M;
int map[MAX][MAX];
int visited[MAX][MAX];
int answer=987654321;

void Calc(){
	int sum=0;
	for(int i=0;i<N;i++){
		for(int j=0;j<N;j++){
			if(map[i][j]==1){
				q.push({i,j});
				int visit[MAX][MAX]={0,};
				visit[i][j]=1;
				int dist=987654321;
				bool flag=false;
				while(!q.empty()){
					auto now=q.front(); q.pop();
					int y=now.first; int x=now.second;
					for(int k=0;k<4;k++){
						int ny=y+dy[k]; int nx=x+dx[k];
						if(ny<0 || nx<0 || ny>N-1 || nx>N-1) continue;
						if(visit[ny][nx]) continue;
						if(!visited[ny][nx] && map[ny][nx]==2){
							if(dist>abs(ny-i)+abs(nx-j)){
								dist=abs(ny-i)+abs(nx-j);
							}
							flag=true;
						}
						else{
							q.push({ny,nx});
							visit[ny][nx]=1;
						}
					}
					if(flag){
						while(!q.empty()) q.pop();
						sum+=dist;
						if(sum>answer) return;
					}
				}
			}
		}
	}
	if(answer>sum) answer=sum;
	return;
}


void Delete(int Count,int Idx){
	
	if(Count==v.size()-M){
		Calc();
		return;
	}
	
	for(int i=Idx;i<v.size();i++){
		int y=v[i].first; int x=v[i].second;
		if(!visited[y][x]){
			visited[y][x]=1;
			Delete(Count+1,i+1);
			visited[y][x]=0;
		}	
	}
}

int main(){
	
	cin >> N >> M;
	
	for(int i=0;i<N;i++){
		for(int j=0;j<N;j++){
			cin >> map[i][j];
			if(map[i][j]==2) v.push_back({i,j});
			
		}		
	}
	
	Delete(0,0);
	
	cout << answer << endl;
	return 0;
}
```

```c++
#include <iostream>
#include <queue>
#include <vector>
#include <algorithm>
#define MAX 51

using namespace std;
vector<pair<int,int>> H;
vector<pair<int,int>> C;

const int dx[]={0,1,0,-1};
const int dy[]={1,0,-1,0};

int N,M;
int map[MAX][MAX];
int visited[MAX][MAX];
int answer=987654321;

void Calc(){
	int sum=0;
	
	for(int i=0;i<H.size();i++){
		int dist=987654321;
		for(int j=0;j<C.size();j++){
			int H_y=H[i].first; int H_x=H[i].second;
			int C_y=C[j].first; int C_x=C[j].second;
			if(visited[C_y][C_x]) continue;
			dist=min(dist,abs(H_y-C_y)+abs(H_x-C_x));
		}
		sum+=dist;
	//	if(sum>answer) return;
	}
	if(answer>sum) answer=sum;
	return;
}


void Delete(int Count,int Idx){
	
	if(Count==C.size()-M){
		Calc();
		return;
	}
	
	for(int i=Idx;i<C.size();i++){
		int y=C[i].first; int x=C[i].second;
		if(!visited[y][x]){
			visited[y][x]=1;
			Delete(Count+1,i+1);
			visited[y][x]=0;
		}	
	}
}

int main(){
	
	cin >> N >> M;
	
	for(int i=0;i<N;i++){
		for(int j=0;j<N;j++){
			cin >> map[i][j];
			if(map[i][j]==2) C.push_back({i,j});
			if(map[i][j]==1) H.push_back({i,j});
			
		}		
	}
	
	Delete(0,0);
	
	cout << answer << endl;
	return 0;
}
```
