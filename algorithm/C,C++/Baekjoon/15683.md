#include <iostream>
#include <algorithm>
#include <string.h>
#define MAX 8
using namespace std;

struct INFO{
	int y,x,dir;
}d[MAX];

const int dy[]={0,1,0,-1},dx[]={1,0,-1,0};
int N,M,cam,res=987654321;
int map[MAX][MAX],backup[MAX][MAX],visit[MAX];
void print(){
	cout << endl;
	for(int i=0;i<N;i++){
		for(int j=0;j<M;j++){
			cout << map[i][j];
			if(map[i][j]==0) cout << " ";
		}
		cout << endl;
	}
	
}

void count(){
	int count=0;
	for(int i=0;i<N;i++){
		for(int j=0;j<M;j++){
			if(map[i][j]==0) count++;
		}
	}
	print();
	res=min(res,count);
}

void Checking(int y,int x,int dir){
	while(1){
		y+=dy[dir]; x+=dx[dir];
		if(y<0 || x<0 || y>N-1 || x>M-1 || map[y][x]==6) return;
		if(map[y][x]==0 || map[y][x]==-1) map[y][x]=-1;
		else continue;
	}
}

void Camera(){
	for(int i=0;i<cam;i++){
		int y=d[i].y,x=d[i].x,dir=d[i].dir;
		int type=map[y][x];
		if(type==1){
			Checking(y,x,dir);
		}
		else if(type==2){
			Checking(y,x,dir);
			Checking(y,x,(dir+2)%4);
		}
		else if(type==3){
			Checking(y,x,dir);
			Checking(y,x,(dir+1)%4);
			Checking(y,x,(dir+2)%4);
		}
		else if(type==4){
			Checking(y,x,dir);
			Checking(y,x,(dir+1)%4);
			Checking(y,x,(dir+2)%4);
		}
		else if(type==5){
			Checking(y,x,dir);
			Checking(y,x,(dir+1)%4);
			Checking(y,x,(dir+2)%4);
			Checking(y,x,(dir+3)%4);
		}
	}
	count();
	memcpy(map,backup,sizeof(backup));
}

void dfs(int cnt){
	if(cnt==cam) return;
	Camera();
	for(int i=0;i<4;i++){
		int tmp=d[cnt].dir;
		d[cnt].dir=(d[cnt].dir+1)%4;
		dfs(cnt+1);
		d[cnt].dir=tmp;
	}
}

int main(){
	cin >> N >> M;

	for(int i=0;i<N;i++){
		for(int j=0;j<M;j++){
			cin >> map[i][j];
			backup[i][j]=map[i][j];
			if(map[i][j]==6 || map[i][j]==0) continue;
			d[cam].y=i; d[cam].x=j; d[cam].dir=0; cam++;			
		}
	}
	dfs(0);
	cout << res << endl;
	return 0;	
}
