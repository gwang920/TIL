# 백준 - 17471 게리멘더링

- 카테고리 : 완전탐색 +BFS
  - 1) 팀 경우의 수를 모두 만든다
  - 2) 만든 팀이 조건을 만족하며 연결되어 있는지 확인한다
  - 3) 연결 됨 = > 두 지역의 차이를 구하고 min값 갱신
  - ​     연결 안 됨 => RETURN

- 2) 의 조건을 깔끔하게 짜려다 시간이 좀 걸렸다. 그냥 queue를 이용해 짰다.
  - 간단하게 연결요소를 확인하는 방법에 대한 고민,,

* 지역변수는 무조건 초기화 하자

#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
#define MAX 11

using namespace std;

vector<int> Supply;
int N;
int map[MAX][MAX];
int visited[MAX]={0,};
int answer=987654321;

void Check(){
  	vector<int> A,B;
  
	  for(int i=1;i<=N;i++){
	  	if(visited[i]) A.push_back(i);
	  	else B.push_back(i);
	  }
	  
	  int visit_A[MAX],visit_B[MAX]; // 수정************************
	  
	  queue<int> Qa,Qb;
	  
	 
	  		Qa.push(A[0]);
	  		visit_A[A[0]]=1;
	  		while(!Qa.empty()){
	  			int now=Qa.front(); Qa.pop();
	  			for(int i=0;i<N;i++){
	  				if(visit_A[i]) continue;
	  				if(map[now][i] && visited[i]){
	  					Qa.push(i);
	  					visit_A[i]=1;
					  }	
				  }
			  }
	 
	  
		  for(int i=1;i<=N;i++){
		  	if(visited[i]!=visit_A[i]) return;
		  }
	  
	 
	  		Qb.push(B[0]);
	  		visit_B[B[0]]=1; ;
	  		while(!Qb.empty()){
	  			int now=Qb.front(); Qb.pop();
	  			for(int i=0;i<N;i++){
	  				if(visit_B[i]) continue;
	  				if(map[now][i] && !visited[i]){
	  					Qa.push(i); // 수정************************
	  					visit_B[i]=1;
					  }	
				  }
			  }
	  
	  for(int i=1;i<=N;i++){
	  	if(visited[i]==visit_B[i]) return;
	  }
	  
	  int SUM_A=0,SUM_B=0;
	  for(int i=0;i<A.size();i++){
	  	SUM_A+=Supply[A[i]-1];
	  }
	  for(int i=0;i<B.size();i++){
	  	SUM_B+=Supply[B[i]-1];
	  }
		  cout << SUM_A << " " <<SUM_B <<endl;
	  
	  answer=min(answer,abs(SUM_A-SUM_B));
	  
}

void Div(int Idx,int Count){
	
	if(Count==5){ // 수정************************
		return;
	}

	for(int i=Idx;i<=N;i++){
		visited[i]=1;
		Check();
		Div(i+1,Count+1);
		visited[i]=0;
	}
}

int main(){
	
	cin >> N;
	
	for(int i=0;i<N;i++){
		int tmp;
		cin >> tmp;
		Supply.push_back(tmp);

	for(int i=1;i<=N;i++){
		int tmp;
		cin >> tmp;
		for(int j=0;j<tmp;j++){
			int x;
			cin >> x;
			map[i][x]=1; map[x][i]=1;
		}
	}

	Div(1,0);
	
	if(answer==987654321){
		cout << "-1" << endl;
	}else{
		cout << answer << endl;
	}
	
	return 0;
}
