# 백준 - 14500 테트로미노

- 카테고리 : 완전탐색
- 모든 경우의 수를 탐색하고, 최대값을 배출하는 문제
- 각 도형의 움직임을 추적하는 함수를 만든다
- 구상은 간단한 문제인데 구현에서 세부적인 요소들 때문에 시간이 지연되었다.





# 성공

```c++
#include <iostream>
#include <algorithm>

using namespace std;

int N,M;
int map[501][501];
int maxi=0;
const int dx[]={1,0,-1,0};
const int dy[]={0,1,0,-1};

struct TETRO{
	
	int count=0;
	int x,y;

	void A(int dir){
			int ny=y, nx=x;
			count=map[ny][nx];
			for(int i=0;i<3;i++){
				nx+=dx[dir]; ny+=dy[dir];
				if(nx<0 || ny<0 || nx>M-1 || ny>N-1) return;
				count+=map[ny][nx];
			}
			maxi=max(maxi,count);
	
		return;
	}
	
	void B(){
		int ny=y; int nx=x;
		count=0;
		for(int i=ny;i<ny+2;i++){
			for(int j=nx;j<nx+2;j++){
				if(i<0 || j<0 || j>M-1 || i>N-1) return;
				count+=map[i][j];
			}
		}
		maxi=max(maxi,count);
		return;
	}
	
	void C(int dir){
		int rot=2;
		int dir2=3;
		while(rot--){
			int ny=y; int nx=x;
			count=map[ny][nx];
			for(int i=0;i<3;i++){
				if(i==2){
					ny+=dy[(dir+dir2)%4]; nx+=dx[(dir+dir2)%4];
					if(ny<0 || nx<0 || ny>N-1 || nx>M-1) break;
					count+=map[ny][nx];	
				}
				else {
					ny+=dy[dir]; nx+=dx[dir];
					if(ny<0 || nx<0 || ny>N-1 || nx>M-1) break;
					count+=map[ny][nx];	
				}
			}
			maxi=max(maxi,count);
			dir2=1;
		}
		return;
	}
	
	void D(int dir){
		int rot=2;
		int dir2=3;
		while(rot--){
			int ny=y; int nx=x;
			count=map[ny][nx];
			for(int i=0;i<3;i++){
				if(i==1){
					ny+=dy[(dir+dir2)%4]; nx+=dx[(dir+dir2)%4];
					if(ny<0 || nx<0 || ny>N-1 || nx>M-1) break;
					count+=map[ny][nx];	
				}else{
					ny+=dy[dir]; nx+=dx[dir];
					if(ny<0 || nx<0 || ny>N-1 || nx>M-1) break;
					count+=map[ny][nx];	
				}
			}
			maxi=max(maxi,count);
			dir2=1;
		}
		return;
	}
	void E(int dir){
		int rot=2;
		int dir2=1;
		while(rot--){
			int ny=y; int nx=x;
			count=map[ny][nx];
			for(int i=0;i<3;i++){
				if(i==1){
					ny+=dy[(dir+dir2)%4]; nx+=dx[(dir+dir2)%4];
					if(ny<0 || nx<0 || ny>N-1 || nx>M-1) break;
					count+=map[ny][nx];	
					ny-=dy[(dir+dir2)%4]; nx-=dx[(dir+dir2)%4];
				}else{
					ny+=dy[dir]; nx+=dx[dir];
					if(ny<0 || nx<0 || ny>N-1 || nx>M-1) break;
					count+=map[ny][nx];	
				}
			}
			maxi=max(maxi,count);
			dir2=3;
		}
		return;
	}
};

int main(){
	
	
	cin >> N >> M;
	
	for(int i=0;i<N;i++){
		for(int j=0;j<M;j++){
			cin >> map[i][j];
		}
	}
	
	TETRO tetro;
	
	for(int i=0;i<N;i++){
		for(int j=0;j<M;j++){
				tetro.y=i; tetro.x=j;
				tetro.B();
			for(int dir=0;dir<4;dir++){
				tetro.A(dir);
				tetro.C(dir);
				tetro.D(dir);
				tetro.E(dir);
			}
		}
	}
	
	cout << maxi << "\n";	
	return 0;
}
```
