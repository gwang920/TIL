# 백준 4485 - 녹색 옷 입은 애가 젤다지?

- 카테고리 : BFS/다익스트라

```
BFS로 탐색하되 매번 모든 노드(모든 배열)에서 가장 최소값만을 갱신한다.

=> 우선순위큐로 (합계,y좌표,x좌표)를 넣어주고 가장 작은 값이 top에 위치하도록 한다.
이렇게 우선순위 큐를 구성하면 매번 방문처리를 하면 가장 작은 값으로 갱신할 수 있다.
```





# 실패

```C++
#include <stdio.h>
#include <queue>

#define _CRT_SECURE_NO_WARNINg

using namespace std;

queue<pair<int, int>> q;

const int dx[4] = { 0,1,0,-1 };
const int dy[4] = { 1,0,-1,0 };


int visited[125][125] = {(0,0),};
int map[125][125];
int backup[125][125] = {(0,0),};
int n;
int flag = 0;

int cnt = 1;


void reset() {

	for (int i = 0;i < n;i++)
	{
		for (int j = 0;j < n;j++) {
			visited[i][j] = 0;
			backup[i][j] = 0;
		}
	}

}

int main(void) {


	while (true) {

		int sum = 0,flag=0;
		int dir_sum=0, dir_y=0, dir_x=0;
		int min_1 = 9999;

		scanf_s("%d", &n);

		if (n == 0) break;
		
		for (int i = 0;i < n;i++) {
			for (int j = 0;j < n;j++) {

				scanf_s("%d", &map[i][j]);
			}
		}

		reset();


		q.push({ 0,0 });
		visited[0][0] = 1;
		dir_sum = map[0][0];

		while (!q.empty()) {
			
			auto now = q.front();
			q.pop();
			int y = now.first;
			int x = now.second;

			if ((y == n - 1 && x == n - 2) || (y == n - 2 && x == n - 1)) {

				dir_sum += map[n - 1][n - 1];
				break;

			}

			for (int i = 0;i < 4;i++) {

				int ny = y + dy[i];
				int nx = x + dx[i];

				if (ny<0 || ny>n - 1 || nx<0 || nx>n - 1) continue;

				if (!visited[ny][nx]) {
					
					visited[ny][nx] = 1;
					sum = dir_sum + map[ny][nx];
			
					if (min_1 > sum) {
						min_1 = sum;
						dir_y = ny;
						dir_x = nx;
						flag = 1;
					}
			}	
		}
			
			if (flag) {
				flag = 0;
				dir_sum = min_1;
				
				backup[dir_y][dir_x] = dir_sum;

				q.push({ dir_y,dir_x });

			}
			min_1 = 9999;
		
		}

		for (int i = 0;i < n;i++) {

			for(int j=0;j<n;j++){
				
				printf("%d  ", backup[i][j]);
				
				}
			printf("\n");
		}

		printf("\n");
		for (int i = 0;i < n;i++) {

			for (int j = 0;j < n;j++) {

				printf("%d  ", visited[i][j]);

			}
			printf("\n");
		}
		printf("\n");
		printf("problem %d: %d\n", cnt,dir_sum);
		printf("\n");
		cnt++;
	}
	return 0;
}
```

# 성공

```C++
#include <iostream>
#include <queue>
#include <string.h>
#include <functional>
#include <climits>
#define MAX 126
using namespace std;

typedef pair<int,pair<int,int>> INFO;
priority_queue<INFO,vector<INFO>,greater<INFO>> pq;
const int dy[]={0,1,0,-1},dx[]={1,0,-1,0};
int N,map[MAX][MAX],dist[MAX][MAX],res=INT_MAX;
bool visit[MAX][MAX];
int main(){
	int cnt=0;
	while(1){
		cin >> N;
		if(N==0) break;
		cnt++;
		for(int i=0;i<N;i++){
			for(int j=0;j<N;j++){
				cin >> map[i][j];
				dist[i][j]=INT_MAX;
			}
		}
		memset(visit,0,sizeof(visit));
		visit[0][0]=1;
		pq.push({map[0][0],{0,0}});
		while(!pq.empty()){
			auto now=pq.top(); pq.pop();
			int sum=now.first,y=now.second.first,x=now.second.second;
			for(int i=0;i<4;i++){
				int ny=y+dy[i],nx=x+dx[i],nsum=sum+map[ny][nx];
				if(ny<0 || nx<0 || ny>N-1 || nx>N-1) continue;
				
				if(!visit[ny][nx] && dist[ny][nx]>nsum){
					dist[ny][nx]=nsum; visit[ny][nx]=1;
					pq.push({nsum,{ny,nx}});
				}
				if(ny==N-1 && nx==N-1){
					res=dist[N-1][N-1];
					break;
				}	
			}
		}
		cout << "Problem " << cnt << ": " << res << endl;
	}
	return 0;
}
```
