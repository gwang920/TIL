#include <iostream>
#include <algorithm>
#include <vector>
#define MAX 20
using namespace std;

vector<int> v;

int N;
int map[MAX][MAX];
int Total,Start,Link;
int answer=987654321;

void dfs(int cnt,int sum,int idx){
	
	if(cnt==N/2){
		int temp;
		temp=abs(Total-sum);
		cout << Total << " ";
		if(answer>abs(temp-Total)){
			answer=abs(temp-Total);
		}
		return;
	}
	
	for(int i=idx+1;i<v.size();i++){
		sum+=map[i];
		dfs(cnt+1,sum,i+1);
		sum-=v[i];
	}
}


int main(){
	
	cin >> N;
	
	for(int i=0;i<N;i++){
		for(int j=0;j<N;j++){
			int tmp;
			cin >> tmp;
			map[i][j]+=tmp;
			map[j][i]+=tmp;
			Total+=tmp;
		}
	}

	for(int i=0;i<N/2;i++){
		dfs(0,0,i);
	}

	cout << answer << endl;
	return 0;
}
