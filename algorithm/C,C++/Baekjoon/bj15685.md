# 성공

```c++
#include <iostream>
#include <vector>
#define MAX 101
using namespace std;

vector<int> v;
const int dy[]={0,-1,0,1};
const int dx[]={1,0,-1,0};

int N;
int answer=0;
int map[MAX][MAX];
int main(){
	cin >> N;
	for(int i=0;i<N;i++){
		int x,y,d,g;
		cin >> x >> y >> d >> g;
		v.push_back(d);
		map[y][x]=1; map[y+dy[d]][x+dx[d]]=1;
		y+=dy[d];x+=dx[d];
		while(g--){
			int size=v.size();
			for(int i=size-1;i>=0;i--){
				int dir=(v[i]+1)%4;
				y+=dy[dir]; x+=dx[dir];
				map[y][x]=1;
				v.push_back(dir);
			}
		}
		v.clear();
	}

	for(int i=0;i<=100;i++){
		for(int j=0;j<=100;j++){
				if(i+1>=101 || j+1>=101) continue;
				if(map[i][j]){
					if(map[i+1][j] && map[i][j+1] && map[i+1][j+1]){
						answer++;
					}
				}
		}
	}
	
	cout << answer << endl;
	return 0;
}
```
