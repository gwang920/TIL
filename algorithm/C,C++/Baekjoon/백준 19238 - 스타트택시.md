```
고려하지 못한 조건
1) 연료가 다 떨어지면 종료한다
2) 가장 짧은 경로를 찾을 때 BFS탐색을 해야한다
3) 택시 시작점에서 손님 출발지, 손님 출발지에서 손님 도착지로 가는 경로가 모두 벽으로 막힌 경우


런타임에러
1. 배열에 할당된 크기를 넘어서 접근했을 때
2. 전역 배열의 크기가 메모리 제한을 초과할 때
3. 지역 배열의 크기가 스택 크기 제한을 넘어갈 때
4. 0으로 나눌 떄
5. 라이브러리에서 예외를 발생시켰을 때
6. 재귀 호출이 너무 깊어질 때
7. 이미 해제된 메모리를 또 참조할 때
8. void, main이 아닌 함수의 return 값이 없을 때
9. main이 0이 아닌 값을 retrun 할 때 (exit(1)도 마찬가지)
```

# 실패

```c++
#include <iostream>
#include <vector>
#include <queue>
#include <string.h>
using namespace std;

struct CUS{
	int sy,sx,gy,gx;
};

struct TAXI{
	int y,x,f;
};

vector<CUS> cu;

const int dy[]={0,1,0,-1};
const int dx[]={1,0,-1,0};

int fuel,N,M,nty,ntx,res;
int board[21][21],cusLoc[21][21];

void cusLocation(){
	memset(cusLoc,0,sizeof(cusLoc));
	for(int i=0;i<cu.size();i++){
		int y=cu[i].sy,x=cu[i].sx;
		cusLoc[y][x]=i+1;
	}
}


bool isMindist(int sel_y,int sel_x,int sel_f,int y,int x,int f){
	if(sel_f<=f){
		if(sel_f<f){
			return true;
		} 
		else if(sel_f==f){
				if(sel_y>=y){
					if(sel_y>y){
						return true;
					}
					else if(sel_y==y){
						if(sel_x>x){
							return true;
						}
					}	
				}
			}
		}
	return false;
}


int selectCus(int ty,int tx){
	cusLocation();
	queue<TAXI> t;
	bool visit[21][21];
	memset(visit,0,sizeof(visit)); visit[ty][tx]=1;
	if(cusLoc[ty][tx]) return cusLoc[ty][tx]-1;
	int sel_y=21,sel_x=21,sel_f=-1;
	int cus_idx=-1;
	t.push({ty,tx,fuel});
	while(!t.empty()){
		TAXI now=t.front(); t.pop();
		int y=now.y,x=now.x,f=now.f;
		
        if(sel_f!=-1 && sel_f>f) continue;
		for(int i=0;i<4;i++){
			int ny=y+dy[i],nx=x+dx[i];
			if(ny<1 || nx<1 || ny>N || nx>N) continue;
			if(visit[ny][nx] || board[ny][nx]) continue;
			visit[ny][nx]=1;
			if(cusLoc[ny][nx]){
				if(isMindist(sel_y,sel_x,sel_f,ny,nx,f)){
					sel_f=f; sel_y=ny; sel_x=nx; cus_idx=cusLoc[ny][nx]-1;
					fuel=f-1; nty=ny; ntx=nx;
                    if(fuel<0){
					cout <<-1;
					exit(0);
					}
				}
				continue;
			}
			t.push({ny,nx,f-1});
        }
	}
	
	return cus_idx;
}

int goalDist(int ty,int tx,int gy,int gx){
     bool visit[21][21];
     memset(visit,0,sizeof(visit));
     visit[ty][tx]=1;
     
     queue<TAXI> q;
     q.push({ty,tx,0});
     while(!q.empty()){
        TAXI now=q.front();q.pop();
        int y=now.y,x=now.x,f=now.f;
        if(f<=0){
        	cout << -1;
        	exit(0);
		}
        for(int i=0;i<4;i++){
         int ny=y+dy[i],nx=x+dx[i];
         if(ny<1 || nx<1 || ny>N || nx>N) continue;
	 	 if(visit[ny][nx] || board[ny][nx]) continue;
	 	 visit[ny][nx]=1;
         if(ny==gy && nx==gx){
           nty=gy;ntx=gx;
           return f-1;
         }
         q.push({ny,nx,f-1});
        }
     }
}

void move(){
	while(cu.size() && fuel){
		int idx=selectCus(nty,ntx);
        fuel+=goalDist(nty,ntx,cu[idx].gy,cu[idx].gx)*2;
		cu.erase(cu.begin()+idx);
	}
	if(cu.size() && !fuel) res=-1;
	res=fuel;
}

void input(){
	cin >> N >> M >> fuel;
	for(int i=1;i<=N;i++){
		for(int j=1;j<=N;j++){
			cin >> board[i][j];
		}
	}
	cin >> nty >> ntx;
	for(int i=0;i<M;i++){
		int a,b,c,d;
		cin >> a >> b >> c >> d;
		cu.push_back({a,b,c,d});
	}
}

int main(){
	input();
	move();
	cout << res << endl;
	return 0;
}
```
