- 모든 경우에 대해서 전부 탐색하고 + 문제에서 요구하는 조건에 따라 실행되도록 코드 작성한다.
- Q. 변수 깔끔하게 정리하는 방법 ?







# 성공

```c++
#include <stdio.h>
#include <algorithm>
#include <iostream>
using namespace std;


int N,M,D;
int map[16][16]={0,};
int backup[16][16]={0,};
int ansb=0;
int maxi=0;
bool flag=false; 
// 초기화 
void init(){
	for(int i=0;i<N;i++){
		for(int j=0;j<M;j++){
			map[i][j]=backup[i][j];
		}
	}
	return;
}

// 전진
void advance(){
	
	for(int i=N-1;i>=0;i--){
		for(int j=M-1;j>=0;j--){
			if(i==0){
			map[i][j]=0;
			continue;	
			} 
			if(map[i-1][j]) flag=true;
			map[i][j]=map[i-1][j]; 
		}
	}
	return;
}

// 공격 (적 지우기)
void attack(int a,int b,int c){
	
	while(1){
		
		int y1=16,x1=16,y2=16,x2=16,y3=16,x3=16;
		int min_a=16,min_b=16,min_c=16;
		for(int i=N-1;i>0;i--){
			for(int j=0;j<M;j++){
				if(map[i][j]){
					int dis_a=(abs(N-i)+abs(a-j));
					int dis_b=(abs(N-i)+abs(b-j));
					int dis_c=(abs(N-i)+abs(c-j));
					
					if(D>=dis_a && min_a>=dis_a){
						if(min_a>dis_a){ // 작으면 무조건 바꿔주고  
							min_a=dis_a;
							y1=i; x1=j;	  
						}
						else if(min_a==dis_a){ // 같다면 가장 왼쪽놈을 선택
							if(x1>j){
								y1=i; x1=j;
							} 	
						}
					}
					if(D>=dis_b && min_b>=dis_b){
						if(min_b>dis_b){
							min_b=dis_b;
							y2=i; x2=j;	  
						}
						else if(min_b==dis_b){
							if(x2>j){
								y2=i; x2=j;
							} 	
						}
					}
					if(D>=dis_c && min_c>=dis_c){
						if(min_c>dis_c){
							min_c=dis_c;
							y3=i; x3=j;	  
						}
						else if(min_c==dis_c){
							if(x3>j){
								y3=i; x3=j;
							} 	
						}
					}
				}
			}
		}

		
		if(y1 || x1){
			if(map[y1][x1]) ansb++;
			map[y1][x1]=0;	
		} 
		if(y2 || x2){
			if(map[y2][x2]) ansb++;
			map[y2][x2]=0;		
		} 
		if(y3 || x3){
			if(map[y3][x3]) ansb++;
			map[y3][x3]=0;		
		}
	
		advance();
		if(flag==false) break;
		flag=false;
		
	}
	return; 
}


int main(){
	
	scanf("%d %d %d",&N,&M,&D);
	
	
	for(int i=0;i<N;i++){
		for(int j=0;j<M;j++){
			int tmp;
			scanf("%1d",&tmp);
			map[i][j]=tmp;
			backup[i][j]=tmp;
		}
	}
	// 모든 경우 탐색
	for(int i=0;i<M;i++){
		for(int j=i+1;j<M;j++){
			for(int k=j+1;k<M;k++){
				ansb=0;
				attack(i,j,k);
				maxi=max(ansb,maxi);
				init();	
			}
		}
	}
	
	printf("%d",maxi);
	
	return 0;
}
```
