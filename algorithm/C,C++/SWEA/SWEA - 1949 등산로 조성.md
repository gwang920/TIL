# SWEA - 1949 등산로 조성

- 카테고리 : 완전탐색/백트래킹
- 간단한 문제였다.



# 성공

```c++
#include <iostream>
#include <vector>
#include <string.h>
#define MAX 8

using namespace std;

struct INFO{
	int y,x,depth;
};
vector<INFO> cmd;

const int dy[]={0,1,0,-1};
const int dx[]={1,0,-1,0};

int T,N,K,map[MAX][MAX],res;
bool visited[MAX][MAX];

void bridge(int i,int j,int cnt){
	res=max(res,cnt);
	for(int dir=0;dir<4;dir++){
		int y=i+dy[dir]; int x=j+dx[dir];
		if(y<0 || x<0 || y>N-1 || x>N-1) continue;
		if(map[y][x]>=map[i][j] || visited[y][x]) continue;
		visited[y][x]=1;
		bridge(y,x,cnt+1);
		visited[y][x]=0;
	}
}

void dig(){
	for(int i=0;i<N;i++){
		for(int j=0;j<N;j++){
			for(int k=1;k<=K;k++){
				map[i][j]-=k;
				for(int start=0;start<cmd.size();start++){
					if(cmd[start].y==i && cmd[start].x==j) continue;	
					bridge(cmd[start].y,cmd[start].x,1);
				}
				map[i][j]+=k;
			}
		}
	}	
}

int main(){
	cin >> T;
	for(int tc=1;tc<=T;tc++){
		cin >> N >> K;
		for(int i=0;i<N;i++){
			for(int j=0;j<N;j++){
				cin >> map[i][j];
				if(cmd.empty()) cmd.push_back({i,j,map[i][j]});
				if(cmd[0].depth<=map[i][j]){
					if(cmd[0].depth<map[i][j]){
						cmd.clear();
						cmd.push_back({i,j,map[i][j]});
					}else{
						cmd.push_back({i,j,map[i][j]});
					}
				}
			}
		}
		dig();
		cout << "#" << tc << ' ' << res << endl;
		cmd.clear();
		res=0; memset(visited,0,sizeof(visited));
	}	
	return 0;
}
```
