```C++
int convert(string s){
	int num[21]={0,};
	int idx=0;
	for(int i=idx;i<s.length();i++){
		if(s[i]>='0' && s[i]<='9'){
			num[idx]*=10;
			num[idx]+=(s[i]-'0');
		}
		if(s[i]=='+' || s[i]=='-' || s[i]=='*' || s[i]=='/') idx++;
	}
	int number=num[0];
	idx=1;
	for(int i=0;i<s.length();i++){
		if(s[i]=='+'){
			if(number+num[idx]>1000) return 0;
			number=number+num[idx];
			idx++;
		}
		if(s[i]=='-'){
			if(number-num[idx]<0) return 0;
			number=number-num[idx];
			idx++;
		} 
		if(s[i]=='*'){
			if(number*num[idx]>=1000) return 0;
			number=number*num[idx];
			idx++;
		}
		if(s[i]=='/'){
			if(num[idx]==0) return 0;
			number/=num[idx];
			idx++;
		} 
	}
	return number;
}
void dfs(int cnt,string s,bool check[21]){

				dfs(cnt+2,s+'*'+num,check);

				dfs(cnt+2,s+'/'+num,check);
			}
```

```c++
#include <iostream>
#include <string.h>
using namespace std;

bool number[10],cmd[5],visit[1000],flag;
int T,N,O,M,goal,res; 

bool check(int i){
	int length,num=i;
	if(i<10) length=1;
	else if(i>=100) length=3;
	else length=2;
	for(int j=0;j<length;j++){
		if(!number[num%10]) return false;
		num/=10;
	}
	return true;
}

void dfs(int cnt,int s){
	if(cnt>M || cnt>=res) return;
	if(s==goal && !flag){
		res=min(res,cnt+1); return;
	}
	flag=false;
 	if(visit[s]) return;
	visit[s]=1;
	for(int i=0;i<=999;i++){
		if(!check(i)) continue;
		int Ncnt=0;
		if(i<10) Ncnt=cnt+1;
		else if(i>=10 && i<=99) Ncnt=cnt+2;
		else Ncnt=cnt+3;
		if(i==goal){
			res=min(res,Ncnt-cnt); return;
		}
		for(int j=1;j<=4;j++){
			if(!cmd[j]) continue;
			int s=s;
			if(j==1){
				if(s+i<1000) dfs(Ncnt+1,s+i);
			}
			if(j==2){
				if(s-i>=0) dfs(Ncnt+1,s-i);
			}
			if(j==3){
				if(s==0) s=1;
				if(s*i<1000){
					dfs(Ncnt+1,s*i);	
				} 
			}
			if(j==4){
				if(s==0) dfs(Ncnt,i);
				else if(i!=0) dfs(Ncnt+1,s/i);
			}
		}	
	}
}

int main(){
	cin >> T;
	int tmp;
	for(int i=1;i<=T;i++){
		cin >> N >> O >> M;
		for(int j=0;j<N;j++){
			cin >> tmp;
			number[tmp]=1;
		}
		for(int j=0;j<O;j++){
			cin >> tmp;
			cmd[tmp]=1;
		}
		cin >> goal;
		res=987654321; flag=true;
		dfs(-1,0);
		if(res==987654321) res=-1;
		cout << "#" << i << " " << res << endl;
		memset(number,0,sizeof(number));
		memset(cmd,0,sizeof(cmd));
		memset(visit,0,sizeof(visit));
	}
	return 0;
}
```

# 성공

```c++
#include <iostream>
using namespace std;

bool number[10],cmd[5],visited[1000],flag;
int visit[1000];
int T,N,O,M,goal,res; 

void init(){
	for(int i=0;i<10;i++) number[i]=0;
	for(int i=0;i<5;i++) cmd[i]=0;
	for(int i=0;i<1000;i++){
		visit[i]=0; visited[i]=0;	
	} 
}
void dfs(int cnt,int s){
	if(cnt>=M || cnt>=res) return;
	if(s==goal && !flag){
		res=min(res,cnt+1); return;
	}
	flag=false;
 	if(visit[s]<=cnt && visit[s]) return;
	visit[s]=cnt;
	for(int i=0;i<=999;i++){
		if(!visited[i]) continue;
		int Ncnt=0;
		if(i<10) Ncnt=cnt+1;
		else if(i>=100) Ncnt=cnt+3;
		else Ncnt=cnt+2;
		if(i==goal){
			res=min(res,Ncnt-cnt); return;
		}
		if(s==0){dfs(Ncnt,i); continue;}
		for(int j=1;j<=4;j++){
			if(!cmd[j]) continue;
			if(j==1){if(s+i<1000) dfs(Ncnt+1,s+i);}
			if(j==2){if(s-i>=0) dfs(Ncnt+1,s-i);}
			if(j==3){if(s*i<1000){dfs(Ncnt+1,s*i);}}
			if(j==4){if(i!=0) dfs(Ncnt+1,s/i);}
		}	
	}
}

void make(int sum){
	if(sum>999) return;
	for(int i=0;i<10;i++){
		if(!number[i]) continue;
		int num=sum*10+i;
		if(num>999) continue;
		if(visited[num]) continue;	
		visited[num]=1;
		make(num);
	}
}

int main(){
	cin >> T;
	int tmp;
	for(int i=1;i<=T;i++){
		cin >> N >> O >> M;
		for(int j=0;j<N;j++){
			cin >> tmp; number[tmp]=1;
		}
		for(int j=0;j<O;j++){
			cin >> tmp; cmd[tmp]=1;
		}
		cin >> goal;
		res=987654321; flag=true;
		make(0); dfs(0,0);
		if(res==987654321) res=-1;
		cout << "#" << i << " " << res << endl;
		init();
	}
	return 0;
}
```
