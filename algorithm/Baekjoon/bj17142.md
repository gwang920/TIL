```c++
#include <iostream>
#include <queue>
#include <vector>
#include <algorithm>
#define MAX 987654321
using namespace std;

queue<pair<int,int>> q;
vector<pair<int,int>> v;

const int dx[]={0,1,0,-1};
const int dy[]={1,0,-1,0};

int N,M;
int map[50][50];
int visited[50][50];
int Select[10];
int blank;
int time=MAX;
int check=0,tmp=1;
bool flag=false;

void init(){
	
	for(int i=0;i<N;i++)
		for(int j=0;j<N;j++)
			visited[i][j]=map[i][j];
			
	return;
}

void Case_calc(int M,int virus){
	
	// 경우의 수  
		int M_backup=M; int virus_backup=virus;
		while(M_backup--){
			tmp*=virus_backup;
			virus_backup--;
		}
		M_backup=M;
		
		while(M--){
			tmp/=M_backup;
			M_backup--;
		}
	
		return;
}


int func(int bla){
	
	int last=0;
	while(!q.empty()){
		
		auto now=q.front(); q.pop();
		int y=now.first; int x=now.second;

		for(int i=0;i<4;i++){
			int ny=y+dy[i]; int nx=x+dx[i];
			
			if(ny<0 || nx<0 || ny>N-1 || nx>N-1) continue;
			
			if(!visited[ny][nx]){
				bla--;
				visited[ny][nx]=visited[y][x]+1;
				last=max(last,visited[ny][nx]);
				q.push({ny,nx});
			}
		}	
	}
	
	if(bla>0){
		check++;
		return MAX;
	}
	
	if(bla==0) return last-2;
}

void Select_Virus(int Depth,int idx){
	
	if(Depth==M){
		for(int i=0;i<v.size();i++){
			if(Select[i]){
				q.push({v[i].first,v[i].second});
			}
		}
		init();
		time=min(func(blank),time);
		return;
	} 
	
	for (int i = idx; i < v.size(); i++)
    {
        if (Select[i] == true) continue;
        Select[i] = true;
        Select_Virus(Depth + 1, i + 1);
        Select[i] = false;
    }
}


int main(){
	
	cin >> N >> M;
	int virus=0;
	for(int i=0;i<N;i++){
		for(int j=0;j<N;j++){
			cin >> map[i][j];
			if(!map[i][j]) blank++;	
			if(map[i][j]==2){
				virus++;
				v.push_back({i,j});
			}
		}
	}
	// 바이러스 개수 = 0 
	if(!blank){
		cout << 0;
		return 0;
	}
	
	Case_calc(M,virus);
	Select_Virus(0,0);
	
	if(check==tmp){
		cout << "-1" << "\n";	
	}else{
		cout << time << "\n";
	}
	return 0;
}
```

```c++
queue<pair<int,int>> q;
vector<pair<int,int>> v;

int visited[50][50];
int Select[10];
int blank;
int time=MAX;
int check=0,tmp=1;
bool flag=false;

		for(int j=0;j<N;j++)

}

void Case_calc(int M,int virus){
	
	// 경우의 수  
		int M_backup=M; int virus_backup=virus;
		while(M_backup--){
			tmp*=virus_backup;
			virus_backup--;
		}

int func(int bla){
	
	int last=0;
	while(!q.empty()){
		
		auto now=q.front(); q.pop();
		int y=now.first; int x=now.second;

		for(int i=0;i<4;i++){
```
