# 백준 4485(녹색 옷 입은 애가 젤다지?)

https://www.acmicpc.net/problem/4485



##### 실패(진행중)


```c++
#include <stdio.h>
#include <queue>

#define _CRT_SECURE_NO_WARNINg

using namespace std;

queue<pair<int, int>> q;

const int dx[4] = { 0,1,0,-1 };
const int dy[4] = { 1,0,-1,0 };


int visited[125][125] = {(0,0),};
int map[125][125];
int backup[125][125] = {(0,0),};
int n;
int flag = 0;

int cnt = 1;


void reset() {

	for (int i = 0;i < n;i++)
	{
		for (int j = 0;j < n;j++) {
			visited[i][j] = 0;
			backup[i][j] = 0;
		}
	}

}

int main(void) {


	while (true) {

		int sum = 0,flag=0;
		int dir_sum=0, dir_y=0, dir_x=0;
		int min_1 = 9999;

		scanf_s("%d", &n);

		if (n == 0) break;
		
		for (int i = 0;i < n;i++) {
			for (int j = 0;j < n;j++) {

				scanf_s("%d", &map[i][j]);
			}
		}

		reset();


		q.push({ 0,0 });
		visited[0][0] = 1;
		dir_sum = map[0][0];

		while (!q.empty()) {
			
			auto now = q.front();
			q.pop();
			int y = now.first;
			int x = now.second;

			if ((y == n - 1 && x == n - 2) || (y == n - 2 && x == n - 1)) {

				dir_sum += map[n - 1][n - 1];
				break;

			}

			for (int i = 0;i < 4;i++) {

				int ny = y + dy[i];
				int nx = x + dx[i];

				if (ny<0 || ny>n - 1 || nx<0 || nx>n - 1) continue;

				if (!visited[ny][nx]) {
					
					visited[ny][nx] = 1;
					sum = dir_sum + map[ny][nx];
			
					if (min_1 > sum) {
						min_1 = sum;
						dir_y = ny;
						dir_x = nx;
						flag = 1;
					}
			}	
		}
			
			if (flag) {
				flag = 0;
				dir_sum = min_1;
				
				backup[dir_y][dir_x] = dir_sum;

				q.push({ dir_y,dir_x });

			}
			min_1 = 9999;
		
		}

		for (int i = 0;i < n;i++) {

			for(int j=0;j<n;j++){
				
				printf("%d  ", backup[i][j]);
				
				}
			printf("\n");
		}

		printf("\n");
		for (int i = 0;i < n;i++) {

			for (int j = 0;j < n;j++) {

				printf("%d  ", visited[i][j]);

			}
			printf("\n");
		}
		printf("\n");
		printf("problem %d: %d\n", cnt,dir_sum);
		printf("\n");
		cnt++;
	}
	return 0;
}
```
