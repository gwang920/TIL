# 백준 - 12100 2048(easy)

- 카테고리 : 완전탐색(?) + 시뮬레이션(?)
- 성공 코드에서 네 방향 탐색 로직을 참고했다.
- cur , next 두 개의 변수로 나누어 map이 회전하는 경우에도 모든 경우를 탐색할 수 있다.
  - temp[i] [j] = map[N-j-1] [i] 코드를 사용하면 오른쪽으로 90도 회전이 가능하다.
- 또, flag를 사용해서 블록이 합쳐질 때, 중복됨을 방지할 수 있다.
- map == 0 일때 continue 해주고 , target이라는 변수를 통해 가장 자리 부터 블록을 채워넣을 수 있다.



# 실패

```c++
#include <iostream>
#include <algorithm>
using namespace std;

int N;
int map[21][21];
int backup[21][21];

void init(){
	
	for(int i=0;i<N;i++)
		for(int j=0;j<N;j++)
			map[i][j]=backup[i][j];
	
	 return;
}

// 빈공간 제거 
void move(int y,int x){
	int dist=0;
	int flag=0; 
	// 최초 빈 공간이 아닌 지점 찾기 
	for(int i=y;i<N;i++){
		if(map[i][x]){
			dist=i-y;			
			flag=1;
			break;
		}
	}
	// 땡겨주기 
	if(!flag) return;
 	for(int i=y;i<N;i++){
		map[i][x]=map[i+dist][x];
	}
	
	return;
}

void sum(){
	
	for(int i=0;i<N;i++){
		for(int j=0;j<N;j++){
			
			// 빈 공간이면
			if(!map[j][i]){
				move(j,i);
	
			}
			if(!map[j+1][i]){
				move(j+1,i);
			} 
		
			// 같으면 
			if(map[j][i]==map[j+1][i]){
				map[j][i]*=2; map[j+1][i]=0;	
			}
			
		}		
	}
	return;
}


void dfs(int dir,int count){
	
	if(count==5) return;
	sum(dir);
	
	for(int i=0;i<4;i++){
		dfs((dir+i)%4,count+1);
	}
	
}

int check(){
	
	int maxx=0;
	for(int i=0;i<N;i++){
		for(int j=0;j<N;j++){
			maxx=max(maxx,map[i][j]);	
		}
	}
	
	return maxx;
}

int main(){
	
	cin >> N;
	for(int i=0;i<N;i++){
		for(int j=0;j<N;j++){
			int tmp;
			cin >> tmp;
			map[i][j]=tmp;
			backup[i][j]=tmp;	
		}
	}
	
	int maxi=0;
	for(int i=0;i<4;i++){
		dfs((dir+i)%4,count+1);
		maxi=max(maxi,check());
		init();
	}
	
	for(int i=0;i<N;i++){
		for(int j=0;j<N;j++){
			cout << map[i][j];
		}
		cout << endl;
	}
	
	printf("%d",maxi);
	return 0;
}
```

# 성공

```c++
#include <stdio.h>
#include <algorithm>

using namespace std;

int N,ret;

struct BOARD{
	int map[20][20];
	
	void rotate(){
		int temp[20][20];
		
		for(int i=0;i<N;i++){
			for(int j=0;j<N;j++){
				temp[i][j]=map[N-j-1][i];
			}
		}
		
		for(int i=0;i<N;i++){
			for(int j=0;j<N;j++){
				map[i][j]=temp[i][j];
			}
		}
		return;
	}
	
	int maxvalue(){
		int tmp=0;
		for(int i=0;i<N;i++){
			for(int j=0;j<N;j++){
				tmp=max(tmp,map[i][j]);
			}
		}
		return tmp;
	}
	
	void up(){
		int temp[20][20];
		
		for(int i=0;i<N;i++){
			int flag=0,target=-1;
			for(int j=0;j<N;j++){
				if(!map[j][i]) continue;
				
				if(flag==1 && temp[target][i]==map[j][i]){
					flag=0;
					temp[target][i]*=2;
				}
				else{
					flag=1;
					temp[++target][i]=map[j][i];
				}
			}
			
			for(++target;target<N;++target){
				temp[target][i]=0;	
			}
		}
		
		for(int i=0;i<N;i++){
			for(int j=0;j<N;j++){
				map[i][j]=temp[i][j];
			}
		}
		return;
	}
};


void dfs(BOARD cur,int count){
	if(count==5){
		ret=max(ret,cur.maxvalue());
		return;
	}	
	
	for(int dir=0;dir<4;dir++){
		BOARD next=cur;
		next.up();
		dfs(next,count+1);
		cur.rotate();
	}
}

int main(){
	
	scanf("%d",&N);
	
	BOARD board;
	for(int i=0;i<N;i++){
		for(int j=0;j<N;j++){
			scanf("%d",&board.map[i][j]);
		}
	}
	
	ret=0;
	dfs(board,0);
	printf("%d\n",ret);
	return 0;
}
```
