- 첫 코드의 DFS를 활용한 탐색방법

  - 메모리초과 + 시간초과

  ```
  이유 : 1) 5x5 부터 map 전체를 탐색 -> 4x4전체 탐색 -> ... (x)
  		map이 1인 지점이 나오면 5 -> 4 -> 3 -> 2 -> 1 부분 탐색해야한다.
   	   2) visited 배열 사용 => 백트래킹으로 해결 가능
  ```

# 실패

```c++
#include <stdio.h>
#include <algorithm>

using namespace std;

int map[10][10]={0,};
int visited[10][10]={0,};
int total=26;
int check_sum[6];


void init(){
	
	for(int i=0;i<10;i++){
		for(int j=0;j<10;j++){
			visited[i][j]=0;
		}
	}
	
	for(int i=1;i<=5;i++){
		check_sum[i]=0;
	}
	
	return;
}

int total_check(){
	
	for(int i=0;i<5;i++){
		if(check_sum[i]>5){
			return 0;
		}
	}
	
	return 1;	
}
int size_check(int y,int x,int size){
	
	for(int i=y;i<=y+size;i++){
		for(int j=x;j<=x+size;j++){
			if(!map[i][j] || visited[i][j]){
				return 0;
			}
		}
	}
	return 1;
}

void visit(int y,int x,int size){
	
	for(int i=y;i<=y+size;i++){
		for(int j=x;j<=x+size;j++){
			visited[i][j]=1;
		}
	}
	return;
}

int dfs(int y, int x,int size,int sum){
	
	
	if(size==-1) return sum;
	if(sum>25 || sum>total) return 99;
//	if(total_check()==0) return 99;
	
	for(int i=y;i<10;i++){
		for(int j=x;j<10;j++){
			if(visited[i][j]) continue;
			
			if(map[i][j]){
				if(i+size<10 && j+size<10){
					if(size_check(i,j,size)){
						sum++;
						check_sum[size]++;
						visit(i,j,size);
					}
					
				}
				
			}
		}
	}
	
	for(int i=0;i<y;i++){
		for(int j=0;j<x;j++){
			if(visited[i][j]) continue;
			if(map[i][j]){
				if(i+size<10 && j+size<10){
					if(size_check(i,j,size)){
						sum++;
						check_sum[size]++;
						visit(i,j,size);
					}
				}
				
			}
		}
	}
	if(sum<total){
		dfs(y,x,size-1,sum);	
	}
	
}

int main(){
	
	for(int i=0;i<10;i++){
		for(int j=0;j<10;j++){
			scanf("%d",&map[i][j]);
		}
	}
	int flag=0;
	for(int i=0;i<10;i++){
		for(int j=0;j<10;j++){
			if(map[i][j]){
				flag=1;
				int tmp=dfs(i,j,4,0);
				if(total_check()){
					total=min(total,tmp);
				}
				init();
			}
		}
	}
	if(!flag) total=0;
	else{
		if(total==26) total=-1;	
	} 
	
	printf("%d\n",total);
	
	return 0;
}
```

## 성공

```c++
#include <stdio.h>
#include <algorithm>

using namespace std;



int map[10][10];
int sum[6]={0,5,5,5,5,5};
int total;
int result;
void dfs(int y, int x){
	
	
	if(x==10){
		dfs(y+1,0);
		return;
	}
	
	if(y==10){
		result=min(result,total);
		return;
	}
	
	if(!map[y][x]){
		dfs(y,x+1);
		return;
	}
	
	
	for(int len=5;len>0;len--){
		
		if(sum[len]==0 || len+y>10 || len+x>10) continue;
		bool flag=true;
		for(int i=0;i<len;i++){
			for(int j=0;j<len;j++){
				
				if(!map[i+y][j+x]){
					flag=false;
					break;	
				}
			}
			if(flag==false){
				break;
			}
		}
		
		if(flag==false){
			continue;
		}	
		
		sum[len]--;
		total++;
		
		for(int i=0;i<len;i++){
			for(int j=0;j<len;j++){
				map[i+y][j+x]=0;
			}
		}
		
		dfs(y,x+len);
		
		sum[len]++;
		total--;
		
		for(int i=0;i<len;i++){
			for(int j=0;j<len;j++){
				map[i+y][j+x]=1;
			}
		}
	}
}



int main(){
	
	for(int i=0;i<10;i++){
		for(int j=0;j<10;j++){
			scanf("%d",&map[i][j]);
		}
	}
	
	result=26;
	dfs(0,0);
	if(result==26){
		printf("-1");
	}else{
		printf("%d",result);
	}
	
	
	return 0;
}
```
