# 성공

```C++
#include <stdio.h>
#include <queue>
using namespace std;

queue<pair<int,int>> q;
queue<pair<int,int>> rotate;
int N,k,L;
int map[101][101]={0,};
int dy=0,dx=1;
int ny=1,nx=1;
int answer=0;
int flag=0;
int tail_y=1;
int tail_x=1;
int main(){
	
	scanf("%d %d",&N,&k);
	
	for(int i=0;i<k;i++){
		int y,x;
		scanf("%d %d",&y,&x);
		map[y][x]=1;
	}
	map[1][1]=2;
	q.push({1,1});

	scanf("%d",&L);
	for(int i=0;i<L;i++){
		int cnt;
		char rot;
		scanf("%d %c",&cnt,&rot);
		rotate.push({cnt,rot});
		}
		while(1){
			ny+=dy;
			nx+=dx;
			answer++;
			if(map[ny][nx]==2 || ny<1 || nx<1 || ny>N || nx>N){
				break;
			}
			
			if(map[ny][nx]==1){
				q.push({ny,nx});
				map[ny][nx]=2;
			} 
			else if(map[ny][nx]==0){
				tail_y=q.front().first;
				tail_x=q.front().second;
				
				map[tail_y][tail_x]=0;
				q.pop();
				q.push({ny,nx});
				map[ny][nx]=2;
			}
		// 방향 전환
		if(answer==rotate.front().first){
			if(rotate.front().second=='D'){
			if(dy==0 && dx==1){
				dy=1; dx=0; 
			}
			else if(dy==0 && dx==-1){
				dy=-1; dx=0;
			}
			else if(dy==1 && dx==0){
				dy=0; dx=-1;
			}
			else if(dy==-1 && dx==0){
				dy=0; dx=1;
			}
		} 
		if(rotate.front().second=='L'){
			if(dy==0 && dx==1){
				dy=-1; dx=0; 
			}
			else if(dy==0 && dx==-1){
				dy=1; dx=0;
			}
			else if(dy==1 && dx==0){
				dy=0; dx=1;
			}
			else if(dy==-1 && dx==0){
				dy=0; dx=-1;
			}
		}
		rotate.pop();
		}
	}


	printf("%d",answer);
	
	return 0;
}
```



# 성공(방향 전환코드 정리)

```c++
#include <stdio.h>

int map[101][101];
int head_y,head_x,tail_index;
int snake_y[10101],snake_x[10101];
int dy[]={0,1,0,-1};
int dx[]={1,0,-1,0};
int main(){

    int n,k,l;
    scanf("%d %d",&n,&k);
    for(int i=0;i<k;i++){
        int a,b;
        scanf("%d%d",&a,&b);
        map[a][b]=1;
    }
    scanf("%d",&l);
    int time;
    int cmd[10001];
    for(int i=0;i<l;i++){
        int time,b;
        scanf("%d %c",&time,&b);
        cmd[time]=b;
    }

    time=0;
    int dir=0;
    head_y=1,head_x=1,tail_index=time;
    snake_y[time]=head_y,snake_x[time]=head_x;
    map[head_y][head_x]=-1;
    while(true){
        time++;
        head_y+=dy[dir];
        head_x+=dx[dir];
        snake_y[time]=head_y;
        snake_x[time]=head_x;
        

        if(head_x<1 || head_x>n || head_y<1 || head_y>n || map[head_y][head_x]==-1) break;

        if(map[head_y][head_x]==0)
        {
            int tail_y=snake_y[tail_index];
            int tail_x=snake_x[tail_index];
            tail_index++;
            map[tail_y][tail_x]=0;
        }
        map[head_y][head_x]=-1;

        if(cmd[time]=='D'){
            dir=(dir+1)%4;
        }
        if(cmd[time]=='L'){
            dir=(dir+3)%4;
        }
    }

    printf("%d",time);


    return 0;
}
```
