```c++
#include <iostream>
#include <queue>
#include <functional>
#define Max 11

using namespace std;
typedef pair<pair<int,int>,int> PAIR;
priority_queue<PAIR,vector<PAIR>,greater<PAIR>> Tree;
priority_queue<PAIR,vector<PAIR>,greater<PAIR>> Backup_Tree;

int N,M,K;
int Food[Max][Max];
int A[Max][Max];
int Backup_A[Max][Max];

const int dx[]={-1,0,1,-1,1,-1,0,1};
const int dy[]={-1,-1,-1,0,0,1,1,1};

void Winter(){
	
	for(int i=1;i<=N;i++){
		for(int j=1;j<=N;j++){
			Food[i][j]+=A[i][j];
		}
	}
	
	for(int i=1;i<=N;i++){
		for(int j=1;j<=N;j++){
			A[i][j]=Backup_A[i][j];
		}
	}
	
	return;
}

void Atum(int y,int x){
	
	for(int i=0;i<8;i++){
		int ny=y+dy[i]; int nx=x+dx[i];
		if(ny<1 || nx<1 || ny>N || nx>N) continue;
		Tree.push({{ny,nx},1});
	}
	return;
}

void Backup(){
	
	while(!Backup_Tree.empty()){
		auto now=Backup_Tree.top(); Backup_Tree.pop();
		int y=now.first.first; int x=now.first.second;
		int age=now.second;
		Tree.push({{y,x},age});
		if(age%5==0) Atum(y,x);
	}
	return;
}


void Func(){
	
	while(1){
		if(K==0) return;
		while(!Tree.empty()){
			auto now=Tree.top(); Tree.pop();
			int y=now.first.first; int x=now.first.second;
			int age=now.second;
			
			// Spring
			if(Food[y][x]>=age){
				Food[y][x]-=age;
				age++;
				Backup_Tree.push({{y,x},age});
			}else{
				// Summer
				A[y][x]+=age/2;
			}
		}
		Backup();
		Winter();
		K--;
	}
}

int main(){
	
	cin >> N >> M >> K;
	
	for(int i=1;i<=N;i++){
		for(int j=1;j<=N;j++){
			cin >> A[i][j];
			Food[i][j]=5;
			Backup_A[i][j]=A[i][j];
		}
	}
	for(int i=0;i<M;i++){
		int x,y,z;
		cin >> x >> y >> z;
		Tree.push({{x,y},z});
	}
	
	Func();
	cout << Tree.size() << endl;
	
	return 0;
}
```
