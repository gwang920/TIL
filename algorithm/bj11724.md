

# 백준 11724 연결 요소의 개수









```
#include <iostream>
#include <vector>

using namespace std;

vector<int> v[1001];

int N,M;
int U,V;
int cnt=0;
int flag;

int main(){
	
	cin >> N >> M;
	
	for(int k=0;k<M;k++){
		
		cin >> U >> V;
		flag=0;
		if(k==0){
			
			v[k].push_back(U);
			v[k].push_back(V);
			cnt++;
		}
		
		else{			
		for(int i=0;i<k;i++){
		
			for(int j=0;j<v[i].size();j++){
				
				if(v[i][j]==U || v[i][j]==V){
					
					v[i].push_back(U);
					v[i].push_back(V);
					flag=1;
					break;
				}
			
			}
		}
		if(!flag){
			cnt++;
			v[k].push_back(U);
			v[k].push_back(V);
			
		}
		}
	
	}
	cout << cnt;
	
	return 0;
	
}
```
