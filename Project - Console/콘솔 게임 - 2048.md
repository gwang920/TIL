# 콘솔 게임 - 2048

```c++
W - up
S - down
A - left
D - right

위의 상,하,좌,우 조작키를 이용하여 2048의 블록을 생성하는 게임

1) 이동 후 board 갱신 시 system("cls")를 사용하면 깜빡임이 눈에 거슬렸다.
=> SetConsoleCursorPosition를 활용하여 변화되는 인덱스의 값만 변경시켜 깜빡임을 최소화했다.

2) 상,하,좌,우 이동 함수를 전부 생성하지 않고, 상으로 이동하는 로직만 구현했다.
  여기에 rotate() 함수를 이용하여 4방향을 전부 구현할 수 있도록 했다.
  
3) board 이동 후 콘솔을 갱신할 때 1),2) 라인을 추가해줘야한다.
    (0 4 128 0) -> (0, 0, 48, 128)
    와 같이 이동 전의 값이 완전히 지워지지 않는 경우가 있기 떄문

for(int i=0;i<4;i++){

					cout << "    ";          // 2)
					gotoxy(4+8*(i),4+2*(j));
					cout << board[j][i];	
				} 
			}
		}
	}
```

# 소스코드

```c++
#include <iostream>
#include <time.h>
#include <conio.h>
#include <windows.h>

using namespace std;
int step=0; 
int board[4][4];
bool is_generate=false;

void gotoxy(int y,int x)
{
	COORD pos={y,x};
	SetConsoleCursorPosition(GetStdHandle(STD_OUTPUT_HANDLE), pos); 
}
int random_index(){
    int random_idx = rand() % 4;
    return random_idx;
}

int random_element(){
    int random_elm = rand() % 10;
    random_elm = (random_elm==0) ? 4 : 2;
    return random_elm;
}

void generate(){
	int first_i,first_j,second_i,second_j;
	if(step==0){
		while(1){
		first_i=random_index(); first_j=random_index();
		second_i=random_index(); second_j=random_index();
		if(first_i==second_i && first_j==second_j) 
			continue;
		else if(board[first_i][first_j] || board[second_i][second_j]) 
			continue;
		else break;
		}
		int first_elm=random_element();
		int second_elm=random_element();
		board[first_i][first_j]=first_elm;
		board[second_i][second_j]=second_elm;
	}else{
		while(1){
			first_i=random_index(); first_j=random_index();
			if(board[first_i][first_j]) continue;
			else break;
		}
		int first_elm=random_element();
		int second_elm=random_element();
		board[first_i][first_j]=first_elm;
	}
	step++;
	return;
}

bool game_over(){
	for(int i=0;i<4;i++){
		for(int j=0;j<4;j++){
			if(!board[i][j]) return false;
			if(j<3 && board[i][j]==board[i][j+1]) return false;
			if(i<3 && board[i][j]==board[i+1][j]) return false;
		}
	}
	return true;
}

bool is_win(){
	for(int i=0;i<4;i++){
		for(int j=0;j<4;j++){
			if(board[i][j]==2048) return true;
		}
	}
	return false;
}


void display(){
	if(step==1){
		system("cls");
		cout << "************* Welcome - 2048 ****************" << endl;
		cout << "Press w - up / s - down / a - left /d -right " << endl;
		cout << "Press q - exit" << endl;
		cout << "---------------------------------" << endl;
		for(int i=0;i<4;i++){
			for(int j=0;j<4;j++){
				if(j==0) cout << "|";
					cout << "   "; 
					if(board[i][j]) cout << board[i][j];
					else cout << " ";
					cout << "   |"; 
				}
				cout << endl;
				cout << "---------------------------------" << endl;
			}
	}
	else{
		for(int i=0;i<4;i++){
			for(int j=0;j<4;j++){
				if(!board[j][i]){
					gotoxy(2+8*(i),4+2*(j));
					cout <<"     ";	
				} 
				else{
					gotoxy(4+8*(i),4+2*(j));
					cout << "    ";
					gotoxy(4+8*(i),4+2*(j));
					cout << board[j][i];	
				} 
			}
		}
	}
}

void move(){
	for(int j=0;j<4;j++){
		for(int i=0;i<4;i++){
			if(!board[i][j]){
				for(int k=i+1;k<4;k++){
					if(board[k][j]){
						board[i][j]=board[k][j]; board[k][j]=0;
						is_generate=true;
						break;
					}
				}
			}
		}
	}
}
void sum(){
	for(int j=0;j<4;j++){
		for(int i=0;i<3;i++){
			if(!board[i][j]) continue;
			if(board[i][j]==board[i+1][j]){
				board[i][j]*=2; board[i+1][j]=0;
				is_generate=true;
			}
		}
	}
}

void rotate(int cnt,int dir){
	if(dir==0){
		while(cnt--){
		int tmp[4][4];
		for(int i=0;i<4;i++){
			for(int j=0;j<4;j++){
				tmp[i][j]=board[j][4-i-1];
			}
		}
		for(int i=0;i<4;i++){
			for(int j=0;j<4;j++){
				board[i][j]=tmp[i][j];
			}
			}
		}
	}
	else{
		while(cnt--){
		int tmp[4][4];
		for(int i=0;i<4;i++){
			for(int j=0;j<4;j++){
				tmp[j][4-i-1]=board[i][j];
			}
		}
		for(int i=0;i<4;i++){
			for(int j=0;j<4;j++){
				board[i][j]=tmp[i][j];
				}
			}
		}
	}
}

void up(){
	move();sum();move();
}
void right(){
	rotate(1,0);
	up();
	rotate(1,1);
}
void left(){
	rotate(3,0);
	up();
	rotate(3,1);
}
void down(){
	rotate(2,0);
	up();
	rotate(2,1);
}
 
int main()
{
	// cursor OFF 
	CONSOLE_CURSOR_INFO cursorInfo = { 0, };
	cursorInfo.dwSize = 1;
	cursorInfo.bVisible = FALSE;
	SetConsoleCursorInfo(GetStdHandle(STD_OUTPUT_HANDLE), &cursorInfo);
	
    int nKey;
    generate();
    display();
    while (1)
    {	
    	if(game_over()){
    		cout << endl;
    		cout << endl;
    		cout << "************* Game Over ****************"<<endl;
    		break;
		}
        if (_kbhit())
        {	is_generate=false;
            nKey = _getch();
            if (nKey == 'q')
                break;
            switch (nKey)
            {
            case 'w':
            	up(); 
                break;
            case 'a':
            	left(); 
                break;
            case 'd':
            	right();
                break;
            case 's':
            	down();
                break;
            }
            if(is_generate){
            	generate();
            	Sleep(100); display();
			}
            if(is_win()){
    			cout << endl;
    			cout << endl;
    			cout << "************* Clear ****************" << endl;
				break;
			}
        }
    }
    return 0;
}
```
